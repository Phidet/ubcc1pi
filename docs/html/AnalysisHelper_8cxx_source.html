00072 bool AnalysisHelper::PassesMomentumThreshold(const art::Ptr<simb::MCParticle> &particle)
00073 {
00074     // The mapping from PDG code to momentum threshold (GeV)
00075     std::map<int, float> pdgToThresholdMap;
00076     pdgToThresholdMap.emplace(2212, 0.3);
00077     pdgToThresholdMap.emplace(2112, std::numeric_limits<float>::max());
00078 
00079     // If we haven't specified this PDG code then assume the threshold is zero - the particle passes by default
00080     const auto iter = pdgToThresholdMap.find(particle->PdgCode());
00081     if (iter == pdgToThresholdMap.end())
00082         return true;
00083 
00084     // Otherwise insist that the particle has sufficient momentum to pass the threshold
00085     return (particle->P() >= iter->second);
