00241 
00242     if (maxStepSize <= std::numeric_limits<float>::epsilon())

00244 
00245     if (points.empty())
00246         throw cet::exception("ParticleID::GetMostProbabledEdx") << " - The input vector of calorimetry points is empty!" << std::endl;
00247 
00248     // If there is only one point, then this is the most probable
00249     if (points.size() == 1)
00250         return points.front().second;
00251 
00252     // Get the maxiumum and the minimum dEdxs
00253     float max = -std::numeric_limits<float>::max();
00254     float min = std::numeric_limits<float>::max();
00255 
00256     for (const auto &point : points)
00257     {
00258         const auto dEdx = point.second;
00259 
00260         if (dEdx > max)
00261             max = dEdx;
00262 
00263         if (dEdx < min)
00264             min = dEdx;
00265     } 
00266 
00267     const auto range = max - min;
00268 
00269     // Determine the number of steps we will to use
00270     const auto nSteps = std::floor(range / maxStepSize);
00271 
00272     // If the range of the data is smaller than the step size, then just return something in the middle
00273     if (static_cast<int>(nSteps) == 0)
00274         return (max + min) * 0.5f;
00275     
00276     // Use Gaussian kernal density estimation to find the most probable value
00277     const float normalization = 1.f / (std::pow(4 * std::asin(1), 2) * kernalWidth * points.size());
00278     float maxDensity = -std::numeric_limits<float>::max();
00279     float mostProbableValue = -std::numeric_limits<float>::max();
00280     
00281     const auto stepSize = range / nSteps;
00282     for (unsigned int i = 0; i <= nSteps; ++i)
00283     {
00284         const auto dEdxSample = min + i * stepSize;
00285         float density = 0.f;
00286 
00287         for (const auto &point : points)
00288         {
00289             // Get the contribution from this point using a gaussian kernal centered at the sample dEdx
00290             const auto dEdx = point.second; 
00291             density += std::exp(std::pow(dEdx - dEdxSample, 2) / (-2.f * kernalWidth * kernalWidth)) * normalization;
00292         }
00293 
00294         if (density < maxDensity)
00295             continue;
00296 
00297         maxDensity = density;
00298         mostProbableValue = dEdxSample;
00299     }
00300     
00301     return mostProbableValue;
