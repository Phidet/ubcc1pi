00730 geo::View_t EventSelection::GetView(const std::bitset<8> &planeMask) const
00731 {
00732     // Here is a hack to get around a bug in the PID code. Different algorithms have different numbering conventions
00733     const bool usesW = planeMask.test(2) || planeMask.test(7);
00734     const bool usesU = planeMask.test(1) || planeMask.test(6);
00735     const bool usesV = planeMask.test(0) || planeMask.test(5);
00736     
00737     if (usesW && !usesU && !usesV)
00738         return geo::kW;
00739     
00740     if (!usesW && usesU && !usesV)
00741         return geo::kU;
00742

00774 void EventSelection::CombineInductionPlanes(const float yzAngle, const int nHitsU, const int nHitsV, const float uValue, const float vValue, const bool uSet, const bool vSet, float &uvValue, bool &uvSet) const
00775 {
00776     uvValue = -std::numeric_limits<float>::max();
00777 
00778     const float uAngle = -acos(0.5f);
00779     const float vAngle = +acos(0.5f);
00780 
00781     const bool uAngleCut = pow(sin(yzAngle - uAngle), 2) > m_config().Sin2YZAngleCut();
00782     const bool vAngleCut = pow(sin(yzAngle - vAngle), 2) > m_config().Sin2YZAngleCut();
00783     
00784     // Check if we have any information available
00785     const bool uAvailable = uSet && uAngleCut;
00786     const bool vAvailable = vSet && vAngleCut;
00787 
00788     uvSet = (uAvailable || vAvailable) && (nHitsU > 0 || nHitsV > 0);
00789     if (!uvSet)
00790         return;
00791 
00792     const float uWeight = static_cast<float>(uAvailable ? nHitsU : 0);
00793     const float vWeight = static_cast<float>(vAvailable ? nHitsV : 0);
00794     const float totalWeight = uWeight + vWeight;
00795 
00796     // ATTN here total weight must be non-zero due to nHits check above
00797     uvValue = (uWeight*uValue + vWeight*vValue) / totalWeight;
00798 }
