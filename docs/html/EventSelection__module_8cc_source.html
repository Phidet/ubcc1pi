00242 {
00243     if (!muons.empty() || !pions.empty() || !protons.empty())
00244         throw cet::exception("EventSelection::PerformPID") << " - Output vectors for muons, pions and protons were not all empty" << std::endl;
00245     
00246     // No need to do anything if there are no final states supplied
00247     if (finalStates.empty())
00248         return;
00249 
00250     // Split the PFParticles into track and shower like based on the decision made in Pandora
00251     PFParticleVector trackLikes;
00252     this->SelectTracksAndShowers(finalStates, trackLikes, showerLikes);
00253 
00254     // Select the PFParticles that pass the MIP selection
00255     const auto mips = this->SelectMIPs(event, trackLikes, chi2ProtonCut);
00256 
00257     // If there are no mips, then just call everything a proton
00258     if (mips.empty())
00259     {
00260         protons = finalStates;
00261         return;
00262     }
00263     
00264     // Get the muon candidate
00265     const auto muon = this->SelectMuon(event, mips);
00266 
00267     // Populate the output vectors
00268     for (const auto &pfParticle : finalStates)
00269     {
00270         if (pfParticle == muon)
00271         {
00272             muons.push_back(pfParticle);
00273             continue;
00274         }
00275 
00276         // Any non-muon MIP is a pion
00277         if (std::find(mips.begin(), mips.end(), pfParticle) != mips.end())
00278         {
00279             pions.push_back(pfParticle);
00280             continue;
00281         }
00282 
00283         // Any non-MIP track is a proton
00284         if (this->IsTrackLike(pfParticle))
00285         {
00286             protons.push_back(pfParticle);
00287             continue;
00288         }
00289     }
