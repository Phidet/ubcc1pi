634  colors.push_back(PlottingHelper::GetColor(Septenary));
635  colors.push_back(PlottingHelper::GetColor(Octonary));

639  {
640  const auto col = PlottingHelper::GetColor(style);
641 
642  if (std::find(colors.begin(), colors.end(), col) == colors.end())
643  colors.push_back(col);
644  }
645
