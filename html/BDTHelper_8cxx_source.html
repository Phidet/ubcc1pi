385 void BDTHelper::GetROCCurve(const BDTResponseWeightVector &signalResponses, const BDTResponseWeightVector &backgroundResponses, const unsigned int nSamplePoints, std::vector<float> &signalPassingRates, std::vector<float> &backgroundRejectionRates, std::vector<float> &signalPassingRateErrors, std::vector<float> &backgroundRejectionRateErrors)
386 {
387  if (signalResponses.empty() || backgroundResponses.empty())
388  throw std::invalid_argument("BDTHelper::GetROCCurve - input vector of responses is empty");
389 
390  if (nSamplePoints < 2)
391  throw std::invalid_argument("BDTHelper::GetROCCurve - need at least 2 sample points");
392 
393  if (!signalPassingRates.empty() || !backgroundRejectionRates.empty() || !signalPassingRateErrors.empty() || !backgroundRejectionRateErrors.empty())
394  throw std::invalid_argument("BDTHelper::GetROCCurve - output vectors need to be empty");
395 
396  // Define a comparison function for an element of a BDTResponseWeightVector - it returns true if the first response is less than the second
397  const auto &comp = [](const auto &a, const auto &b) { return a.first < b.first; };
398 
399  // Sort the responses
400  auto sortedSignalResponses = signalResponses;
401  std::sort(sortedSignalResponses.begin(), sortedSignalResponses.end(), comp);
402 
403  auto sortedBackgroundResponses = backgroundResponses;
404  std::sort(sortedBackgroundResponses.begin(), sortedBackgroundResponses.end(), comp);
405 
406  // Get the maximum and minimum BDT responses
407  const auto epsilon = std::numeric_limits<float>::epsilon();
408  const auto minX = std::min(sortedSignalResponses.front(), sortedBackgroundResponses.front()).first - epsilon;
409  const auto maxX = std::max(sortedSignalResponses.back(), sortedBackgroundResponses.back()).first + epsilon;
410 
411  // Sample between minX and maxX
412  std::vector<float> vectS, vectB;
413  unsigned int iSignal = 0u;
414  unsigned int iBackground = 0u;
415  float signalWeight = 0.f;
416  float backgroundWeight = 0.f;
417 
418  for (unsigned int iSample = 0; iSample < nSamplePoints; ++iSample)
419  {
420  const auto X = minX + (maxX - minX) * (iSample / static_cast<float>(nSamplePoints - 1));
421 
422  // Sum up the signal and background responses until we find one that is greater than the current cut value
423  while (iSignal < sortedSignalResponses.size())
424  {
425  if (sortedSignalResponses.at(iSignal).first > X)
426  break;
427 
428  signalWeight += sortedSignalResponses.at(iSignal).second;
429  iSignal++;
430  }
431 
432  while (iBackground < sortedBackgroundResponses.size())
433  {
434  if (sortedBackgroundResponses.at(iBackground).first > X)
435  break;
436 
437  backgroundWeight += sortedBackgroundResponses.at(iBackground).second;
438  iBackground++;
439  }
440 
441  // Store the values at this point
442  vectS.push_back(signalWeight);
443  vectB.push_back(backgroundWeight);
444  }
445 
446  // Sanity check that we have seen every BDT response
447  if (iSignal != sortedSignalResponses.size())
448  throw std::logic_error("BDTHelper::GetROCCurve - sanity check failed, haven't used all signal responses");
449 
450  if (iBackground != sortedBackgroundResponses.size())
451  throw std::logic_error("BDTHelper::GetROCCurve - sanity check failed, haven't used all background responses");
452 
453  const auto signalWeightTotal = signalWeight;
454  const auto backgroundWeightTotal = backgroundWeight;
455 
456  // Get the fractions used in the ROC curve
457  for (unsigned int iSample = 0; iSample < nSamplePoints; ++iSample)
458  {
459  // The signal passing rate
460  const auto fracS = 1.f - (vectS.at(iSample) / signalWeightTotal);
461  const auto errS = AnalysisHelper::GetEfficiencyUncertainty(vectS.at(iSample), signalWeightTotal);
462 
463  // The background rejection rate
464  const auto fracB = vectB.at(iSample) / backgroundWeightTotal;
465  const auto errB = AnalysisHelper::GetEfficiencyUncertainty(vectB.at(iSample), backgroundWeightTotal);
466 
467  signalPassingRates.push_back(fracS);
468  backgroundRejectionRates.push_back(fracB);
469 
470  signalPassingRateErrors.push_back(errS);
471  backgroundRejectionRateErrors.push_back(errB);
472  }
473 }

477 float BDTHelper::GetROCIntegral(const std::vector<float> &signalPassingRates, const std::vector<float> &backgroundRejectionRates)
478 {
479  if (signalPassingRates.size() != backgroundRejectionRates.size())
480  throw std::invalid_argument("BDTHelper::GetROCIntegral - sizes of input vectors doesn't match");
481 
482  const auto nSamples = signalPassingRates.size();
483  if (nSamples < 2)
484  throw std::invalid_argument("BDTHelper::GetROCIntegral - need at least 2 samples");
485 
486  // Sort the input data points by signal passing rate
487  std::vector< std::pair<float, float> > data;
488  for (unsigned int i = 0; i < nSamples; ++i)
489  {
490  data.emplace_back(signalPassingRates.at(i), backgroundRejectionRates.at(i));
491  }
492  std::sort(data.begin(), data.end(), [](const auto &a, const auto &b) { return a.first < b.first; });
493 
494  // Perform trapezium integration
495  float integral = 0.f;
496  for (unsigned int i = 1; i < nSamples; ++i)
497  {
498  const auto f1 = data.at(i).first;
499  const auto f0 = data.at(i-1).first;
500  const auto g1 = data.at(i).second;
501  const auto g0 = data.at(i-1).second;
502 
503  const auto df = f1 - f0;
504  const auto dg = g1 - g0;
505 
506  // Calculate the area of the rectangle between f0 -> f1 at height g0
507  const auto rectArea = df * g0;
508 
509  // Calculate the area of the triangle between f0 -> f1 on top of this rectangle made between g0 -> g1
510  const auto triArea = 0.5 * df * dg;
511 
512  // Add to the integral
513  integral += rectArea + triArea;
514  }
515 
516  return integral;
517 }

521 float BDTHelper::GetROCIntegralError(const std::vector<float> &signalPassingRates, const std::vector<float> &backgroundRejectionRates, const std::vector<float> &signalPassingRateErrs, const std::vector<float> &backgroundRejectionRateErrs, const unsigned nUniverses)
522 {
523  const auto nSamples = signalPassingRates.size();
524  if (backgroundRejectionRates.size() != nSamples || signalPassingRateErrs.size() != nSamples || backgroundRejectionRateErrs.size() != nSamples)
525  throw std::invalid_argument("BDTHelper::GetROCIntegralError - sizes of input vectors doesn't match");
526 
527  if (nUniverses <= 1)
528  throw std::invalid_argument("BDTHelper::GetROCIntegralError - need at least two universes");
529 
530  // Create normal distributions for each sample point in the ROC curve
531  std::default_random_engine generator;
532  std::vector< std::normal_distribution<float> > signalDistributions, backgroundDistributions;
533  for (unsigned int i = 0; i < nSamples; ++i)
534  {
535  signalDistributions.emplace_back(signalPassingRates.at(i), signalPassingRateErrs.at(i));
536  backgroundDistributions.emplace_back(backgroundRejectionRates.at(i), backgroundRejectionRateErrs.at(i));
537  }
538 
539  // Here we use a MC technique to estimate the uncertainty
540  std::vector<float> rocIntegrals;
541  float summedROCIntegral = 0.f;
542 
543  for (unsigned int iUniv = 0; iUniv < nUniverses; ++iUniv)
544  {
545  // Generate signal passing rate and background rejection rate vectors with random 1 sigma variations
546  std::vector<float> generatedSignalPassingRates, generatedBackgroundRejectionRates;
547 
548  for (unsigned int i = 0; i < nSamples; ++i)
549  {
550  generatedSignalPassingRates.push_back(signalDistributions.at(i)(generator));
551  generatedBackgroundRejectionRates.push_back(backgroundDistributions.at(i)(generator));
552  }
553 
554  // Now get the ROC integral for this universe
555  const auto rocIntegral = BDTHelper::GetROCIntegral(generatedSignalPassingRates, generatedBackgroundRejectionRates);
556 
557  rocIntegrals.push_back(rocIntegral);
558  summedROCIntegral += rocIntegral;
559  }
560 
561  // Find the standard deviation of the generate ROC integrals
562  const auto mean = summedROCIntegral / static_cast<float>(nUniverses);
563 
564  float summedSquareDifferences = 0.f;
565  for (unsigned int iUniv = 0; iUniv < nUniverses; ++iUniv)
566  {
567  summedSquareDifferences += std::pow(rocIntegrals.at(iUniv) - mean, 2);
568  }
569 
570  const auto variance = summedSquareDifferences / static_cast<float>(nUniverses - 1);
571  const auto uncertainty = std::pow(variance, 0.5f);
572 
573  return uncertainty;
574 }
