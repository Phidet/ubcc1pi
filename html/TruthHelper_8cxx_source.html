17 {
18  art::Ptr<simb::MCTruth> selectedTruth;

22  for (const auto &mcTruth : CollectionHelper::GetCollection<simb::MCTruth>(event, mcTruthLabel))
23  {

205 void TruthHelper::FollowScatters(const art::Ptr<simb::MCParticle> &particle, const MCParticleMap &mcParticleMap, ScatterVector &scatters, art::Ptr<simb::MCParticle> &scatteredParticle)
206 {
207  scatteredParticle = particle;
208 
209  // First look for elastic scatters
210  for (const auto &daughter : TruthHelper::GetDaughters(particle, mcParticleMap))
211  {
212  if (daughter->Process() == "hadElastic")
213  {
214  try
215  {
216  scatters.push_back(TruthHelper::GetElasticScatter(particle, daughter));
217  }
218  catch (const cet::exception &ex)
219  {
220  std::cout << "TruthHelper::FollowScatters. WARNING - " << ex.explain_self() << std::endl;
221  }
222  }
223  }
224 
225  // Now follow inelastic scatters
226  if (particle->EndProcess().find("Inelastic") == std::string::npos)
227  return;
228 
229  MCParticleVector inelasticProducts;
230  art::Ptr<simb::MCParticle> finalStateParticle;
231  bool foundInelasticScatter = false;
232 
233  for (const auto &daughter : TruthHelper::GetDaughters(particle, mcParticleMap))
234  {
235  if (daughter->Process().find("Inelastic") != std::string::npos)
236  {
237  // If the daughter has the same PDG as the incident particle, then assume it's the same particle before & after the scatter
238  if (daughter->PdgCode() != particle->PdgCode())
239  {
240  inelasticProducts.push_back(daughter);
241  continue;
242  }
243 
244  // If there are multiple particles from the inelastic collision, then don't treat it as a scatter
245  if (foundInelasticScatter)
246  {
247  foundInelasticScatter = false;
248  break;
249  }
250 
251  finalStateParticle = daughter;
252  foundInelasticScatter = true;
253  }
254  }
255 
256  if (!foundInelasticScatter)
257  return;
258 
259  // Get the last momentum point before the scatter
260  TVector3 initialMomentum(-std::numeric_limits<float>::max(), -std::numeric_limits<float>::max(), -std::numeric_limits<float>::max());
261  for (unsigned int i = 0; i < particle->NumberTrajectoryPoints(); ++i)
262  {
263  if (particle->T(i) >= finalStateParticle->T())
264  break;
265 
266  initialMomentum = particle->Momentum(i).Vect();
267  }
268 
269  scatters.emplace_back(false, initialMomentum, finalStateParticle->Momentum().Vect(), finalStateParticle, inelasticProducts);
270  TruthHelper::FollowScatters(finalStateParticle, mcParticleMap, scatters, scatteredParticle);
271 }

275 TruthHelper::Scatter TruthHelper::GetElasticScatter(const art::Ptr<simb::MCParticle> &incident, const art::Ptr<simb::MCParticle> &target)
276 {
277  if (target->Process() != "hadElastic")
278  throw cet::exception("TruthHelper::GetElasticScatter") << " - Input target particle doesn't look to be from an elastic scatter" << std::endl;
279 
280  if (target->Mother() != incident->TrackId())
281  throw cet::exception("TruthHelper::GetElasticScatter") << " - Input target particle isn't a daughter of the input incident particle" << std::endl;
282 
283  const unsigned int nTrajectoryPoints = incident->NumberTrajectoryPoints();
284  if (nTrajectoryPoints < 2)
285  throw cet::exception("TruthHelper::GetElasticScatter") << " - Incident particle has less than 2 trajectory points!" << std::endl;
286 
287  // Find the trajectory points just before and just after the scatter
288  const float scatterTime(target->T());
289  unsigned int postScatterIndex = std::numeric_limits<unsigned int>::max();
290  bool foundScatterPoint = false;
291 
292  for (unsigned int i = 1; i < incident->NumberTrajectoryPoints(); ++i)
293  {
294  if (incident->T(i - 1) <= scatterTime && incident->T(i) >= scatterTime && incident->T(i - 1) < incident->T(i))
295  {
296  foundScatterPoint = true;
297  postScatterIndex = i;
298  }
299  }
300  unsigned int preScatterIndex = postScatterIndex - 1;
301 
302  if (!foundScatterPoint)
303  throw cet::exception("TruthHelper::GetElasticScatter") << " - Couldn't find trajectory points before and after the scatter" << std::endl;
304 
305  const auto initialMomentum(incident->Momentum(preScatterIndex).Vect());
306  const auto finalMomentum(incident->Momentum(postScatterIndex).Vect());
307 
308  return TruthHelper::Scatter(true, initialMomentum, finalMomentum, incident, { target });
309 }

313 TruthHelper::EndState TruthHelper::GetEndState(const art::Ptr<simb::MCParticle> &particle, const MCParticleMap &mcParticleMap)
314 {
315  EndState::Type type = EndState::Type::Other;
316  const auto finalParticleMomentum = particle->Momentum(std::max(static_cast<unsigned int>(0), particle->NumberTrajectoryPoints() - 2)).Vect();
317  MCParticleVector products;
318 
319  // ATTN for now, just return OTHER for all non-pions, we only use the end-state of the pions
320  if (particle->PdgCode() != 211)
321  return TruthHelper::EndState(type, finalParticleMomentum, products);
322 
323  // Follow the scatters to make sure we are really looking at the correct MCParticle
324  ScatterVector scatters;
325  art::Ptr<simb::MCParticle> scatteredParticle;
326  TruthHelper::FollowScatters(particle, mcParticleMap, scatters, scatteredParticle);
327  if (particle != scatteredParticle)
328  throw cet::exception("TruthHelper::GetEndState") << " - Input MCParticle undergos a scatter before reaching it's end-state" << std::endl;
329 
330  bool hasPi0 = false;
331  bool hasDecayMuon = false;
332  bool hasDecayMuonNeutrino = false;
333 
334  for (const auto &daughter : TruthHelper::GetDaughters(particle, mcParticleMap))
335  {
336  // Ignore ionisation electrons
337  if (daughter->PdgCode() == 11 && daughter->Process() == "hIoni")
338  continue;
339 
340  // Ignore nuclei from elastic scatters
341  if (daughter->Process() == "hadElastic")
342  continue;
343 
344  // Treat everything else as an end-state interaction product
345  products.push_back(daughter);
346 
347  if (daughter->PdgCode() == 111 && daughter->Process() == "pi+Inelastic")
348  hasPi0 = true;
349 
350  if (daughter->PdgCode() == -13 && daughter->Process() == "Decay")
351  hasDecayMuon = true;
352 
353  if (daughter->PdgCode() == 14 && daughter->Process() == "Decay")
354  hasDecayMuonNeutrino = true;
355  }
356 
357  // Work out the end-state type
358  if (products.empty())
359  {
360  type = EndState::Type::None;
361  }
362  else if (hasDecayMuon && hasDecayMuonNeutrino && products.size() == 2)
363  {
364  type = EndState::Type::DecayToMuon;
365  }
366  else if (particle->EndProcess() == "pi+Inelastic")
367  {
368  type = hasPi0 ? EndState::Type::Pi0ChargeExchange : EndState::Type::InelasticAbsorption;
369  }
370 
371  return TruthHelper::EndState(type, finalParticleMomentum, products);
372 }
