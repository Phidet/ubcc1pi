105 inline void UBFileHelper::ReadMatrixData(const std::string &fileName, std::vector< std::vector<float> > &data)
106 {
107  if (!data.empty())
108  throw std::invalid_argument("UBFileHelper::ReadMatrixData - Vector supplied to accept matrix data isn't empty");
109 
110  // Open the input file
111  std::ifstream file(fileName);
112  if (!file.is_open())
113  throw std::runtime_error("UBFileHelper::ReadMatrixData - Failed to open file \"" + fileName + "\"");
114 
115  // Read the file line-by-line
116  std::string line;
117  size_t lineNumber = 0;
118  while (std::getline(file, line))
119  {
120  lineNumber++;
121 
122  // Trim the line of any preceeding or trailing whitespace
123  line.erase(0, line.find_first_not_of(m_whitespace));
124  line.erase(line.find_last_not_of(m_whitespace) + 1);
125 
126  // Skip empty lines
127  if (line.empty())
128  continue;
129 
130  // Skip comment lines (starting with #)
131  if (line.front() == '#')
132  continue;
133 
134  // Add a new row to the values
135  data.emplace_back();
136  auto &row = data.back();
137 
138  // Extract each value on the line
139  float value;
140  std::stringstream stream(line);
141  while (!stream.eof())
142  {
143  stream >> value;
144 
145  if (stream.fail() || stream.bad())
146  {
147  // Clear any data that's currently been read
148  data.clear();
149 
150  throw std::runtime_error("UBFileHelper::ReadMatrixData - Problem reading line " + std::to_string(lineNumber) + " of \"" +
151  fileName + "\". It must contain only floating point numbers and whitespace. "
152  + "Comments lines begin with # as the first character.");
153  }
154 
155  row.push_back(value);
156  }
157  }
158 }
