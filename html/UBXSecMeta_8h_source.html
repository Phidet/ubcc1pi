87 inline UBXSecMeta::UBXSecMeta(const std::vector<float> &binEdges, const bool hasUnderflow, const bool hasOverflow, const bool scaleByBinWidth) :
88  m_binEdges(binEdges),
89  m_hasUnderflow(hasUnderflow),
90  m_hasOverflow(hasOverflow),
91  m_scaleByBinWidth(scaleByBinWidth)
92 {
93  // Check if the number of bin edges is enough for the underflow, overflow and at least one more bin
94  if (static_cast<int>(binEdges.size()) - 1 <= (hasUnderflow ? 1 : 0) + (hasOverflow ? 1 : 0))
95  throw std::invalid_argument("UBXSecMeta::UBXSecMeta - insufficient bin edges supplied");
96 
97  // Store the number of bins
98  m_nBins = binEdges.size() - 1;
99 
100  // Check that the bin edges are valid
101  for (size_t iBin = 0; iBin < m_nBins; ++iBin)
102  {
103  const auto lowerEdge = binEdges.at(iBin);
104  const auto upperEdge = binEdges.at(iBin + 1);
105  const auto width = upperEdge - lowerEdge;
106 
107  if (width <= std::numeric_limits<float>::epsilon())
108  throw std::invalid_argument("UBXSecMeta::UBXSecMeta - supplied bin " + std::to_string(iBin) + " has width " + std::to_string(width));
109  }
110 }

167 inline UBMatrix UBXSecMeta::GetBinWidths() const
168 {
169  if (!m_scaleByBinWidth)
170  {
171  // If the cross-section wasn't scaled by bin width, then just return 1.f for the width of each bin
172  return ubsmear::UBMatrix(std::vector<float>(m_nBins, 1.f), m_nBins, 1);
173  }
174 
175  std::vector<float> elements;
176  for (unsigned int iBin = 1; iBin < m_binEdges.size(); ++iBin)
177  {
178  const auto lowerEdge = m_binEdges.at(iBin - 1);
179  const auto upperEdge = m_binEdges.at(iBin);
180  const auto width = upperEdge - lowerEdge;
181 
182  if (width <= std::numeric_limits<float>::epsilon())
183  throw std::logic_error("UBXSecMeta::GetBinWidth - Found a bin with zero or negative width");
184 
185  elements.push_back(width);
186  }
187 
188  return ubsmear::UBMatrix(elements, m_nBins, 1);
189 }
