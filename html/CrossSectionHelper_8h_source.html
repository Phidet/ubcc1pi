99  /**
100  * @brief A pair of shared_ptr to matrices:
101  * - first = bias vector
102  * - second = covariance matrix
103  *
104  * For multisim parameters, the bias vector gives the difference between the nominal value in each bin and the mean value
105  * over all universes. The covariance matrix gives the variations of the universes around the mean value. With these
106  * definitions the bias vector elements b_i and covariance matrix elements V_ij can be combined to determine the variations
107  * of the universes around the nominal: E_ij = V_ij + b_i*b_j.
108  *
109  * For unisim parameters, the bias vector gives the difference between the values in each bin when calculated using varied
110  * and central-value samples. This difference is scaled by the ratio of the nominal simulation to central-value sample,
111  * such that it can be compared on the same footing as the multisim parameters. The bias in each bin is interpreted as the
112  * one sigma variation from the nominal, and so can be used to form an error matrix using E_ij = b_i*b_j. For unisim
113  * parameters the covariance matrix (second entry in the pair) is zero by construction.
114  */
115  typedef std::pair< std::shared_ptr<ubsmear::UBMatrix>, std::shared_ptr<ubsmear::UBMatrix> > SystBiasCovariancePair;

917 template <typename R, typename... Args>
918 inline R CrossSectionHelper::SystCacheFunction<R, Args...>::operator()(const std::string &paramName, const unsigned int universeIndex, Args ... args)
919 {
920  // Check that we have a valid parameter name and universe index
921  const auto iterDimensions = m_dimensions.find(paramName);
922  if (iterDimensions == m_dimensions.end())
923  throw std::invalid_argument("SystCacheFunction - Unknown paramName: " + paramName);
924 
925  const auto nUniverses = iterDimensions->second;
926  if (universeIndex >= nUniverses)
927  throw std::out_of_range("SystCacheFunction - Universe index is out of range");
928 
929  // Check if we have a cached entry for this parameter name
930  const auto iterCacheName = m_cache.find(paramName);
931  if (iterCacheName != m_cache.end())
932  {
933  // Check if we have a cached entry for this universe index
934  const auto iterCacheUniverse = iterCacheName->second.find(universeIndex);
935  if (iterCacheUniverse != iterCacheName->second.end())
936  {
937  // We have a cached entry!
938  return iterCacheUniverse->second;
939  }
940  }
941 
942  // We don't have an entry in the cache, so make one
943  const auto result = m_function(paramName, universeIndex, args...);
944  m_cache[paramName][universeIndex] = result;
945 
946  // Return the result
947  return result;
948 }

973  if (systMap.size() != dimensions.size())

975

982
