40  auto getFineBinEdges = [&](const unsigned int nBinsTarget, const float min, const float max, const std::vector<float> &coarseBinEdges) -> std::vector<float> {
41 
42  // If required extend/truncate the bin edges to the min and max
43  std::vector<float> extendedCoarseBinEdges;
44  for (const auto &edge : coarseBinEdges)
45  {
46  if (edge >= min && edge <= max)
47  extendedCoarseBinEdges.push_back(edge);
48  }
49 
50  if (extendedCoarseBinEdges.front() - min > std::numeric_limits<float>::epsilon())
51  extendedCoarseBinEdges.insert(extendedCoarseBinEdges.begin(), min);
52 
53  if (max - extendedCoarseBinEdges.back() > std::numeric_limits<float>::epsilon())
54  extendedCoarseBinEdges.push_back(max);
55 
56  if (!config.makeBinningPlots.useFineBinEdges)
57  {
58  return extendedCoarseBinEdges;
59  }
60 
61  // Now break up the coarse bins into smaller bins
62  // Use the nBinsTarget to determine the ideal width of these smaller bins
63  const auto targetBinWidth = (max - min) / static_cast<float>(nBinsTarget);
64 
65  std::vector<float> fineBinEdges(1, extendedCoarseBinEdges.front());
66  for (unsigned int i = 1; i < extendedCoarseBinEdges.size(); ++i)
67  {
68  const auto lowerEdge = extendedCoarseBinEdges.at(i - 1);
69  const auto upperEdge = extendedCoarseBinEdges.at(i);
70  const auto binWidth = upperEdge - lowerEdge;
71 
72  const auto nFineBins = std::max(1u, static_cast<unsigned int>(std::round(binWidth / targetBinWidth)));
73  const auto fineBinWidth = binWidth / static_cast<float>(nFineBins);
74 
75  // Add the upper edges of the fine bins
76  for (unsigned int j = 1; j <= nFineBins; ++j)
77  {
78  const auto fineUpperEdge = lowerEdge + j * fineBinWidth;
79  fineBinEdges.push_back(fineUpperEdge);
80  }
81  }
82 
83  return fineBinEdges;
84  };
