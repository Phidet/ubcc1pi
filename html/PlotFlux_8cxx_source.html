18 void PlotFlux(const Config &config)
19 {
20  // TODO make configurable
21  const float maxNuEnergy = 6.8f;
22 
23  // Setup a map to hold the flux histograms
24  std::map< int, std::shared_ptr<TH1F> > histMap;
25 
26  // The maximum & minimum flux bins
27  float maxFlux = -std::numeric_limits<float>::max();
28  float minFlux = +std::numeric_limits<float>::max();
29 
30  // Loop over the neutrino PDG codes
31  for (const auto &pdg : {+14, -14, +12, -12})
32  {
33  // Get the flux for this neutrino flavour
34  const auto fluxHistNames = CrossSectionHelper::GetNominalFluxHistNames({ pdg }, config.flux.nuPdgToHistName, config.flux.nomHistPattern);
35  const auto &[fluxBinEdges, fluxValues] = CrossSectionHelper::ReadNominalFlux(config.flux.fileName, fluxHistNames, config.flux.pot);
36 
37  // Setup a histogram for this flux
38  const auto name = fluxHistNames.front() + "_ubcc1pi";
39  const auto nBins = fluxBinEdges.size() - 1;
40  auto pHist = std::make_shared<TH1F>(name.c_str(), "", nBins, fluxBinEdges.data());
41 
42  // Fill the histogram
43  for (unsigned int iBin = 1; iBin <= nBins; ++iBin)
44  {
45  const auto binContent = fluxValues.at(iBin - 1);
46  pHist->SetBinContent(iBin, binContent);
47 
48  // Get the minimum and maximum values (provided this bin isn't zero)
49  if (binContent >= std::numeric_limits<float>::epsilon() && fluxBinEdges.at(iBin) <= maxNuEnergy)
50  {
51  maxFlux = std::max(maxFlux, binContent);
52  minFlux = std::min(minFlux, binContent);
53  }
54  }
55 
56  // Add the hisogram to the map
57  histMap.emplace(pdg, pHist);
58  }
59 
60  // Set the line colours
61  PlottingHelper::SetLineStyle(histMap.at(+14), PlottingHelper::Primary);
62  PlottingHelper::SetLineStyle(histMap.at(-14), PlottingHelper::Primary);
63  PlottingHelper::SetLineStyle(histMap.at(+12), PlottingHelper::Secondary);
64  PlottingHelper::SetLineStyle(histMap.at(-12), PlottingHelper::Secondary);
65 
66  // Used dashed lines for the antineutrinos
67  histMap.at(-14)->SetLineStyle(2);
68  histMap.at(-12)->SetLineStyle(2);
69 
70  // Add a bit of padding to the top of the plot
71  maxFlux += std::exp(std::log(maxFlux - minFlux) * 0.05);
72 
73  // Make the plot
74  auto pCanvas = PlottingHelper::GetCanvas();
75  pCanvas->SetLogy();
76  bool isFirst = true;
77  for (const auto &[pdg, pHist] : histMap)
78  {
79  pHist->GetXaxis()->SetRangeUser(0.f, maxNuEnergy);
80  pHist->GetYaxis()->SetRangeUser(minFlux, maxFlux);
81 
82  if (isFirst)
83  {
84  pHist->Draw("hist");
85  }
86  else
87  {
88  pHist->Draw("hist same");
89  }
90 
91  isFirst = false;
92  }
93 
94  PlottingHelper::SaveCanvas(pCanvas, "plotFlux_allFluxes");
95 }
