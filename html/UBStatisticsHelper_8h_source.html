48 inline std::pair<float, size_t> UBStatisticsHelper::GetChi2(const UBMatrix &smearedPrediction, const UBMatrix &data, const UBMatrix &covarianceMatrix, const float precision)
49 {
50  // Check the input objects have consistent dimensions
51  if (smearedPrediction.GetColumns() != 1u)
52  throw std::invalid_argument("UBStatisticsHelper::GetChi2 - input smeared prediction is not a column vector");
53 
54  const auto nBins = smearedPrediction.GetRows();
55 
56  if (data.GetColumns() != 1u)
57  throw std::invalid_argument("UBStatisticsHelper::GetChi2 - input data is not a column vector");
58 
59  if (data.GetRows() != nBins)
60  throw std::invalid_argument("UBStatisticsHelper::GetChi2 - input data has a different number of bins to the input smeared prediction");
61 
62  if (covarianceMatrix.GetRows() != nBins || covarianceMatrix.GetColumns() != nBins)
63  throw std::invalid_argument("UBStatisticsHelper::GetChi2 - dimensions of the input covariance matrix, doesn't match the input data or smeared prediction");
64 
65  // Insist that the covariance matrix is symmetric
66  if (!UBMatrixHelper::IsSymmetric(covarianceMatrix))
67  throw std::invalid_argument("UBStatisticsHelper::GetChi2 - the input covariance matrix is not symmetric");
68 
69  // Get the eigenvalues and eigenvectors of the covariance matrix
70  const auto &[eigenvalues, eigenvectorMatrix] = UBMatrixHelper::GetEigenDecomposition(covarianceMatrix, precision);
71 
72  // Get the matrix which transforms an input vector into the basis of eigenvectors
73  const auto transformationMatrix = eigenvectorMatrix.GetTranspose();
74 
75  // Transform the smeared prediction and input data into the basis of eigenvectors
76  const auto smearedPredictionEigen = transformationMatrix * smearedPrediction;
77  const auto dataEigen = transformationMatrix * data;
78 
79  // Get the difference between the smeared prediction and the data in the eigenbasis
80  const auto diffEigen = dataEigen - smearedPredictionEigen;
81 
82  // Calculate the test statistic
83  float chi2 = 0.f;
84  size_t dof = 0u;
85 
86  for (size_t iBin = 0; iBin < nBins; ++iBin)
87  {
88  // The eigenvalues of the covariance matrix give the variance along each eigenvector
89  const auto variance = eigenvalues.At(iBin, 0);
90 
91  // Check if the eigenvalue is zero.
92  // This represents a combination of bins in which there is no variance
93  if (std::abs(variance) <= std::numeric_limits<float>::epsilon())
94  {
95  // ATTN here we are not allowing the input covariance matrix to be non-invertible
96  // we could instead just skip this bin, and calculated the test-statistic in the subspace where the covariance matrix is invertible
97  throw std::logic_error("UBStatisticsHelper::GetChi2 - input covariance matrix has a zero eigenvalue");
98  }
99 
100  // Check that we don't have a negative eigenvalue - this should never happen for a true covariance matrix
101  if (variance < 0.f)
102  throw std::logic_error("UBStatisticsHelper::GetChi2 - input covariance matrix has negative eigenvalue: " + std::to_string(variance));
103 
104  // Add to the total chi2 and increase the number of degrees of freedom
105  const auto diff = diffEigen.At(iBin, 0);
106  chi2 += (diff * diff) / variance;
107  dof += 1u;
108  }
109 
110  return {chi2, dof};
111 }

115 inline float UBStatisticsHelper::GetPValue(const float chi2, const size_t degreesOfFreedom, const float precision)
116 {
117  if (precision < std::numeric_limits<float>::epsilon())
118  throw std::invalid_argument("UBStatisticsHelper::GetPValue - input precision is <= 0, it should be 0 -> 1 not inclusive");
119 
120  if (precision >= 1.f)
121  throw std::invalid_argument("UBStatisticsHelper::GetPValue - input precision is >= 1, it should be 0 -> 1 not inclusive");
122 
123  // The cumulative distribution function of the chi2 distribution with k degrees of freedom is given by:
124  //
125  // F(x; k) = g(k/2, x/2) / G(k/2)
126  // - g is the lower incomplete gamma function
127  // - G is the gamma function
128  //
129  // This gives the probability of a test statistic of x or less, to get the p-value we need 1 - F(x; k).
130 
131  // Without any extra dependencies, C++ includes the gamma function from cmath. However, the lower incomplete gamma function isn't
132  // supplied. Here we use a power-series expansion of F(x; k):
133  //
134  // F(x; k) = (x/2)^(k/2) * exp(-x/2) * sum_r^inf [ (x/2)^r / G(k/2 + r + 1) ]
135  //
136  // This equation is simplified by the substitution y = x/2, s = k/2
137 
138  const auto y = chi2 / 2.f;
139  const auto s = static_cast<float>(degreesOfFreedom) / 2.f;
140 
141  // Get the multiplictative factor that's outside of the sum
142  const auto factor = std::pow(y, s) * std::exp(-y);
143 
144  // Define the function for the term that gets summed
145  const auto getTerm = [&](const size_t r)
146  {
147  // ATTN gamma function has no zeros
148  return (std::pow(y, static_cast<float>(r)) / std::tgamma(s + r + 1));
149  };
150 
151  // Do the sum until the value is equal to the last iteration within the precision specified
152  float sum = 0.f;
153  float lastSum = 0.f;
154  size_t r = 0;
155 
156  do
157  {
158  // If we have reached the maximum possible index, then there's no more that we can do
159  if (r == std::numeric_limits<size_t>::max())
160  {
161  std::cerr << "UBStatisticsHelper::GetPValue - WARNING, Unable to calculate p-value, reached maximum number of terms without converging to desired precision" << std::endl;
162  return -std::numeric_limits<float>::max();
163  }
164 
165  lastSum = sum;
166  sum += getTerm(r++);
167 
168  // Check we haven't reached the maximum possible number
169  if (!std::isnormal(sum))
170  {
171  std::cerr << "UBStatisticsHelper::GetPValue - WARNING, Unable to calculate p-value, sum is not converging fast enough" << std::endl;
172  return -std::numeric_limits<float>::max();
173  }
174 
175  }
176  while ( std::abs(sum - lastSum) > precision * sum );
177 
178  // Calculate the p-value
179  const auto pValue = 1.f - (factor * sum);
180 
181  return pValue;
182 }
