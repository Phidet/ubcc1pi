239 {

279 SelectionHelper::EventSelection SelectionHelper::GetDefaultSelection()
280 {
281  // Define the cuts
282  const std::vector<EventSelection::Cut> cuts = {
283  {"passesCCInclusive"},
284  {"min2Tracks"},
285  {"max1Uncontained"},
286  {"2NonProtons", -0.06f},
287  {"pionNotInGap"},
288  {"muonNotInGap"},
289  {"openingAngle", 2.65f},
290  {"topologicalScore", 0.67f},
291  {"startNearVertex", 9.5f},
292  {"likelyGoldenPion", -0.03f}
293  };
294 
295  // Load up the BDTs and store them in a map
296  EventSelection::BDTMap bdtMap = {
297  {"muon", std::make_shared<BDTHelper::BDT>("muon", BDTHelper::MuonBDTFeatureNames)},
298  {"proton", std::make_shared<BDTHelper::BDT>("proton", BDTHelper::ProtonBDTFeatureNames)},
299  {"goldenPion", std::make_shared<BDTHelper::BDT>("goldenPion", BDTHelper::GoldenPionBDTFeatureNames)}
300  };
301 
302  // Define the actual selection logic
303  const auto logic = [](const auto &pEvent, auto &bdtMap, auto &cutTracker)
304  {
305  // ----------------------------------------------------------------------------------
306  // passesCCInclusive
307  // ----------------------------------------------------------------------------------
308 
309  // Insist the event passes the CC inclusive preselection
310  if (!pEvent->reco.passesCCInclusive())
311  return false;
312 
313  // Mark the cut "passesCCInclusive" as passed
314  cutTracker.MarkCutAsPassed("passesCCInclusive");
315 
316  // ----------------------------------------------------------------------------------
317  // min2Tracks
318  // ----------------------------------------------------------------------------------
319 
320  // Count the particles with a track fit and check if they are contained
321  unsigned int nTrackParticles = 0u;
322  unsigned int nUncontainedParticles = 0u;
323 
324  const auto &recoParticles = pEvent->reco.particles;
325  for (unsigned int i = 0; i < recoParticles.size(); ++i)
326  {
327  const auto &particle = recoParticles.at(i);
328 
329  if (!AnalysisHelper::HasTrackFit(particle))
330  continue;
331 
332  nTrackParticles++;
333 
334  if (!AnalysisHelper::IsContained(particle))
335  nUncontainedParticles++;
336  }
337 
338  // Insist that we have at least two tracks
339  if (nTrackParticles < 2)
340  return false;
341 
342  // Mark the cut "min2Tracks" as passed
343  cutTracker.MarkCutAsPassed("min2Tracks");
344 
345  // ----------------------------------------------------------------------------------
346  // max1Uncontained
347  // ----------------------------------------------------------------------------------
348 
349  // Insist that at most one particle is uncontained
350  if (nUncontainedParticles > 1)
351  return false;
352 
353  // Mark the cut "max1Uncontained" as passed
354  cutTracker.MarkCutAsPassed("max1Uncontained");
355 
356  // Identify the muon candidate
357  auto &pMuonBDT = bdtMap.at("muon");
358  const auto muonIndex = SelectionHelper::GetMuonCandidateIndex(recoParticles, BDTHelper::MuonBDTFeatureNames, *pMuonBDT);
359 
360  // Assign the muon candidate a muon PDG code
361  cutTracker.AssignPDGCode(muonIndex, 13);
362 
363  // ----------------------------------------------------------------------------------
364  // 2NonProtons
365  // ----------------------------------------------------------------------------------
366 
367  // Identify the rest of the particles using the proton BDT
368  const auto protonBDTThreshold = cutTracker.GetCutValue("2NonProtons");
369 
370  // Get the proton BDT from the map
371  auto &pProtonBDT = bdtMap.at("proton");
372 
373  // Keep track of the number of protons and pions we have identifies
374  unsigned int nProtons = 0;
375  std::vector<unsigned int> pionIndices;
376 
377  for (unsigned int i = 0; i < recoParticles.size(); ++i)
378  {
379  const auto &particle = recoParticles.at(i);
380 
381  // Skip the muon candidate as we've already identified it
382  if (i == muonIndex)
383  continue;
384 
385  // Assume particles without a track fit are just small protons
386  if (!AnalysisHelper::HasTrackFit(particle))
387  {
388  nProtons++;
389  cutTracker.AssignPDGCode(i, 2212);
390 
391  continue;
392  }
393 
394  // The particle should be contained (as only the muon candidate is allowed to escape). But for sanity, let's check
395  if (!AnalysisHelper::IsContained(particle))
396  throw std::logic_error("DefaultSelection - Found an uncontained particle that isn't the muon. This shouldn't happen!");
397 
398  // Get run the proton BDT
399  std::vector<float> features;
400  const auto hasFeatures = BDTHelper::GetBDTFeatures(particle, BDTHelper::ProtonBDTFeatureNames, features);
401 
402  // If one or more of the BDT features are missing, then assume the particle is a proton
403  if (!hasFeatures)
404  {
405  nProtons++;
406  cutTracker.AssignPDGCode(i, 2212);
407 
408  continue;
409  }
410 
411  // Insist that the BDT response is greater than the cut value to identify the particle as a proton
412  const auto bdtResponse = pProtonBDT->GetResponse(features);
413  if (bdtResponse >= protonBDTThreshold)
414  {
415  nProtons++;
416  cutTracker.AssignPDGCode(i, 2212);
417 
418  continue;
419  }
420 
421  // If we've got to this point then we haven't identified the particle as a muon or a proton.
422  // Instead let's identify the particle as a pion
423  pionIndices.push_back(i);
424  cutTracker.AssignPDGCode(i, 211);
425  }
426 
427  // Sanity check that we have identified every particle
428  const auto nMuons = 1u;
429  const auto nPions = pionIndices.size();
430  if (nProtons + nPions + nMuons != recoParticles.size())
431  throw std::logic_error("DefaultSelection - Identified the wrong number of particles. This shouldn't happen!");
432 
433  // Insist that we exacly one pion (i.e we have have 2 non-protons)
434  if (nPions != 1)
435  return false;
436 
437  // Mark the cut "2NonProtons" as passed
438  cutTracker.MarkCutAsPassed("2NonProtons");
439 
440  // ----------------------------------------------------------------------------------
441  // pionNotInGap
442  // ----------------------------------------------------------------------------------
443 
444  // Get the pion reco particle
445  const auto pionIndex = pionIndices.front();
446  const auto &pion = recoParticles.at(pionIndex);
447 
448  // Sanity check that our muon and pion are not the same particle
449  if (muonIndex == pionIndex)
450  throw std::logic_error("DefaultSelection - The muon and the pion candidates are the same particle. This shouldn't happen!");
451 
452  // Insist that the pion has at least one hit in each view (i.e. not in a gap)
453  if (pion.nHitsU() == 0 || pion.nHitsV() == 0 || pion.nHitsW() == 0)
454  return false;
455 
456  // Mark the cut "pionNotInGap" as passed
457  cutTracker.MarkCutAsPassed("pionNotInGap");
458 
459  // ----------------------------------------------------------------------------------
460  // muonNotInGap
461  // ----------------------------------------------------------------------------------
462 
463  // Get the muon reco particle
464  const auto &muon = recoParticles.at(muonIndex);
465 
466  // Insist that the muon has at least one hit in each view (i.e. not in a gap)
467  if (muon.nHitsU() == 0 || muon.nHitsV() == 0 || muon.nHitsW() == 0)
468  return false;
469 
470  // Mark the cut "muonNotInGap" as passed
471  cutTracker.MarkCutAsPassed("muonNotInGap");
472 
473  // ----------------------------------------------------------------------------------
474  // openingAngle
475  // ----------------------------------------------------------------------------------
476 
477  // Get the opening angle cut value
478  const auto maxOpeningAngle = cutTracker.GetCutValue("openingAngle");
479 
480  // Get the opening angle between the muon and pion
481  const auto muonDir = TVector3(muon.directionX(), muon.directionY(), muon.directionZ()).Unit();
482  const auto pionDir = TVector3(pion.directionX(), pion.directionY(), pion.directionZ()).Unit();
483  const auto openingAngle = muonDir.Angle(pionDir);
484 
485  // Insist that the opening angle isn't too wide
486  if (openingAngle >= maxOpeningAngle)
487  return false;
488 
489  // Mark the cut "openingAngle" as passed
490  cutTracker.MarkCutAsPassed("openingAngle");
491 
492  // ----------------------------------------------------------------------------------
493  // topologicalScore
494  // ----------------------------------------------------------------------------------
495 
496  // Get the topological score cut value
497  const auto minTopologicalScore = cutTracker.GetCutValue("topologicalScore");
498 
499  // Insist that the topological score is above the cut value
500  if (pEvent->reco.selectedTopologicalScore() <= minTopologicalScore)
501  return false;
502 
503  // Mark the cut "topologicalScore" as passed
504  cutTracker.MarkCutAsPassed("topologicalScore");
505 
506  // ----------------------------------------------------------------------------------
507  // startNearVertex
508  // ----------------------------------------------------------------------------------
509 
510  // Get the start near vertex cut value
511  const auto maxVertexDist = cutTracker.GetCutValue("startNearVertex");
512  const auto maxVertexDist2 = maxVertexDist*maxVertexDist;
513 
514  // Insist that all particles with a fitted track start near the vertex
515  const auto recoVertex = pEvent->reco.nuVertex();
516  for (const auto &particle : recoParticles)
517  {
518  // Skip particles without a track fit
519  if (!AnalysisHelper::HasTrackFit(particle))
520  continue;
521 
522  // Get the distance between the particle's start position and the vertex
523  const TVector3 start(particle.startX(), particle.startY(), particle.startZ());
524  const auto vertexDist2 = (start - recoVertex).Mag2();
525 
526  // Insist that this isn't too large
527  if (vertexDist2 > maxVertexDist2)
528  return false;
529  }
530 
531  // Mark the cut "startNearVertex" as passed
532  cutTracker.MarkCutAsPassed("startNearVertex");
533 
534  // ----------------------------------------------------------------------------------
535  // likelyGoldenPion
536  // ----------------------------------------------------------------------------------
537 
538  // Get the likely golden pion cut value
539  const auto goldenPionBDTThreshold = cutTracker.GetCutValue("likelyGoldenPion");
540 
541  // Get the golden pion BDT
542  auto &pGoldenPionBDT = bdtMap.at("goldenPion");
543 
544  // Get the features of the pion
545  std::vector<float> features;
546  if (!BDTHelper::GetBDTFeatures(pion, BDTHelper::GoldenPionBDTFeatureNames, features))
547  throw std::logic_error("DefaultSelection - Can't get golden pion BDT features of pion candidate");
548 
549  // Insist that the BDT response is greater than the cut value to identify the pion as a golden pion
550  const auto bdtResponse = pGoldenPionBDT->GetResponse(features);
551  if (bdtResponse <= goldenPionBDTThreshold)
552  return false;
553 
554  // Mark the cut "likelyGoldenPion" as passed
555  cutTracker.MarkCutAsPassed("likelyGoldenPion");
556 
557  // We passed all cuts!
558  return true;
559  };
560 
561  return EventSelection(cuts, bdtMap, logic);
562 }

573 unsigned int SelectionHelper::GetMuonCandidateIndex(const std::vector<Event::Reco::Particle> &particles, const std::vector<std::string> &featureNames, BDTHelper::BDT &muonBDT)
574 {
575  bool foundCCInclusiveMuon = false;
576 
577  unsigned int ccInclusiveMuonIndex = std::numeric_limits<unsigned int>::max();
578  unsigned int muonIndex = std::numeric_limits<unsigned int>::max();
579 
580  unsigned int nUncontainedParticles = 0;
581 
582  for (unsigned int index = 0; index < particles.size(); ++index)
583  {
584  const auto &particle = particles.at(index);
585 
586  // Check if this particle is the CC inclusive muon candidate
587  if (particle.isCCInclusiveMuonCandidate())
588  {
589  if (foundCCInclusiveMuon)
590  throw std::logic_error("SelectionHelper::GetMuonCandidateIndex - found multiple CC inclusive muon candidates");
591 
592  foundCCInclusiveMuon = true;
593  ccInclusiveMuonIndex = index;
594  }
595 
596  if (!AnalysisHelper::HasTrackFit(particle))
597  continue;
598 
599  // For now make the last escaping particle the muon candidate, if there are multiple we will fall back on the CC inclusive candidate
600  if (!AnalysisHelper::IsContained(particle))
601  {
602  muonIndex = index;
603  nUncontainedParticles++;
604  }
605  }
606 
607  if (!foundCCInclusiveMuon)
608  throw std::logic_error("SelectionHelper::GetMuonCandidateIndex - found no CC inclusive muon candidate");
609 
610  if (nUncontainedParticles > 1)
611  return ccInclusiveMuonIndex;
612 
613  if (nUncontainedParticles == 1)
614  return muonIndex;
615 
616  // If we are here all particles are contained, choose the muon using the BDT
617  float maxMuonBDTResponse = -std::numeric_limits<float>::max();
618  bool foundMuon = false;
619 
620  for (unsigned int index = 0; index < particles.size(); ++index)
621  {
622  const auto &particle = particles.at(index);
623 
624  if (!AnalysisHelper::HasTrackFit(particle))
625  continue;
626 
627  if (!AnalysisHelper::IsContained(particle))
628  throw std::logic_error("SelectionHelper::GetMuonCandidateIndex - found escaping particle when not expecting to!");
629 
630  std::vector<float> features;
631  const auto hasFeatures = BDTHelper::GetBDTFeatures(particle, featureNames, features);
632 
633  if (!hasFeatures)
634  continue;
635 
636  const auto muonBDTResponse = muonBDT.GetResponse(features);
637 
638  if (muonBDTResponse < maxMuonBDTResponse)
639  continue;
640 
641  maxMuonBDTResponse = muonBDTResponse;
642  muonIndex = index;
643  foundMuon = true;
644  }
645 
646  // If no muon can be found, then default to the CC inclusive candidate
647  if (!foundMuon)
648  return ccInclusiveMuonIndex;
649 
650  return muonIndex;
651 }
