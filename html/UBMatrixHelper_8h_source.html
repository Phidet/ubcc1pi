209 inline std::pair<UBMatrix, UBMatrix> UBMatrixHelper::GetEigenDecomposition(const UBMatrix &matrix, const float precision)
210 {
211  // Here we implement the Jacobi eigenvalue algorithm. A summary of this algorithm is described below.
212  //
213  // The intention of this algorithm perfom eigenvalue decomposition, i.e. to express the input matrix, M in diagonal form: M = Q D Q^-1
214  // - D = diagonal matrix whose elements are the eigenvalues of M
215  // - Q = matrix whose columns are the eigenvectors of M
216  // - Q^-1 = the inverse matrix of Q
217  //
218  // If M is a real symmetric matrix (which we insist to be true), then the matrix of eigenvectors, Q, is an orthogonal matrix: Q^-1 = Q^T
219  // - Q^T = the transform matrix of Q (switching rows for columns)
220  //
221  // The Jacobi eigenvalue algorithm is iterative. Each iteration (labelled with n), results in a new matrix D_n. This matrix is formed by
222  // applying a rotation matrix, R_n, to the outcome of the previous iteration: D_n = R_n D_n-1 R_n^T
223  // - D_n = the outcome of the nth iteration of the algorithm
224  // - D_n-1 = the outcome of the (n-1)th iteration of the algorithm
225  // - R_n = the rotation matrix used in the nth iteration
226  // - R_n^T = the transpose of the rotation matrix, R_n
227  // - D_0 = M, the algorithm is seeded with the input matrix
228  //
229  // Note that rotation matrices are orthogonal, and the matrix product of two orthogonal matrices is itself orthogonal. Hence, the matrix
230  // product: (R_1 R_2 R_3 ... R_N) is an orthogonal matrix. The aim of the algorithm is to apply successive rotations such that D_n
231  // converges toward a diagonal matrix. After N iterations, if the sequence has converged, then we can make the approximation:
232  // - D = D_N
233  // - Q^T = R_N ... R_3 R_2 R_1
234  //
235  // In particular, R_n is chosen to be a Givens rotation matrix (i.e. a rotation in a plane defined by two coordinate axes). The plane
236  // chosen to correspond to the off-diagonal element of D_n-1 with the largest absolute value (known as the pivot). The angle of rotation
237  // is chosen such that this pivot element becomes zero after the rotation. The angle is given by:
238  //
239  // tan(2 theta) = 2 [ij] / ([jj] - [ii])
240  // - theta = angle of rotation
241  // - [ij] = the value of D_n-1 in the ith row and jth column. i and j are the row and column of the pivot.
242  //
243  // To determine if D_n has converged to a diagonal matrix, we consider the sum of the squares of its off-diagonal entries, X_n^2. If the
244  // matrix has size s, then there are s(s-1) off-diagonal elements. We define: S = s(s-1)/2, where series of S rotations is known as a
245  // "sweep". One can show that after S rotations, we have:
246  //
247  // X_{n+S} <= r X_n
248  // - X_n = square-root of the sum of the squares of the off-diagonal elements of D_n
249  // - X_{n+S} = the above, after a further S iterations of the algorithm
250  // - r = (1 - 1/S)^(S/2) = the worst-case-scenario linear convergence rate after one sweep
251  // - For large matrices, r tends to e^(1/2), which is approximately 0.61
252  // - S = half the number of the number of off-diagonal elements
253 
254  // Insist that the input matrix is symmetric
255  if (!UBMatrixHelper::IsSymmetric(matrix))
256  throw std::invalid_argument("UBMatrixHelper::GetEigenDecomposition - Input matrix isn't symmetric");
257 
258  // Get the size of the matrix (it is square)
259  const auto size = matrix.GetRows();
260 
261  // If the size is one (i.e. the matrix is just a scalar, then the answer is trivial)
262  if (size == 1)
263  {
264  return std::pair<UBMatrix, UBMatrix> (
265  {{ matrix.At(0, 0) }, 1, 1}, // Eigenvalue is the entry of the matrix
266  {{ 1.f }, 1, 1} // Normalised eigenvector is just the number 1.0
267  );
268  }
269 
270  // Determine the maximum number of iterations we should use - start by assuming one
271  size_t maxIterations = 1;
272  float squaredSumTarget = -std::numeric_limits<float>::max(); // The target square of sum of diagonals to reach the desired precision
273 
274  // ATTN for size 1, the answer is trivial. For size 2, only one iteration is required to get the answer precicely
275  if (size > 2)
276  {
277  // Get the sweep length
278  const size_t sweep = size * (size - 1) / 2;
279 
280  // Get the initial squared sum of the off-diagonals
281  const auto squaredSumInitial = UBMatrixHelper::GetOffDiagonalSquaredSum(matrix);
282 
283  // Get the off-diagonal squared sum we would expect at the target precision
284  squaredSumTarget = 2 * sweep * std::pow(precision, 2.f);
285 
286  // ATTN if we are already within the desired precision, then just run one iteration
287  if (squaredSumInitial > squaredSumTarget && squaredSumInitial > std::numeric_limits<float>::epsilon())
288  {
289  // Get the worst-case-scenario convergence rate for a single sweep
290  const auto convergenceRate = std::pow((1.f - (1.f / static_cast<float>(sweep))), static_cast<float>(sweep) / 2.f);
291 
292  // Work out how many iterations we would need to reach the desired precision in the worst case scenario
293  maxIterations = std::ceil(sweep * std::log(squaredSumTarget/squaredSumInitial) / std::log(convergenceRate));
294  }
295  }
296 
297  // Make a copy of the input matrix which will converge to be diagonal
298  auto diagMatrix = matrix;
299 
300  // Make a matrix to hold the total transformation we applied to the matrix to make it diagonal (start with a unit matrix)
301  auto transformationMatrix = UBMatrixHelper::GetUnitMatrix(size);
302 
303  // Repeat for each iteration of the algorithm
304  for (size_t i = 0; i < maxIterations; ++i)
305  {
306  // Identify the pivot
307  const auto &[pivotRow, pivotCol] = UBMatrixHelper::GetLargestOffDiagonalElement(diagMatrix);
308 
309  // Get the angle through which we should rotate
310  const auto valueRowCol = diagMatrix.At(pivotRow, pivotCol);
311  const auto valueRowRow = diagMatrix.At(pivotRow, pivotRow);
312  const auto valueColCol = diagMatrix.At(pivotCol, pivotCol);
313  const auto theta = std::atan2(2 * valueRowCol, valueColCol - valueRowRow) / 2;
314 
315  // Get the rotation matrix and it's transpose
316  const auto rotationMatrix = UBMatrixHelper::GetGivensRotationMatrix(size, pivotRow, pivotCol, theta);
317  const auto rotationMatrixTranspose = rotationMatrix.GetTranspose();
318 
319  // Apply the rotation
320  diagMatrix = rotationMatrix * diagMatrix * rotationMatrixTranspose;
321  transformationMatrix = rotationMatrix * transformationMatrix;
322 
323  // Get the squared sum of the off-diagonals after this iteration
324  const auto squaredSum = UBMatrixHelper::GetOffDiagonalSquaredSum(diagMatrix);
325 
326  // If we have reached the desired precision, then stop
327  if (squaredSum < squaredSumTarget)
328  break;
329  }
330 
331  // ATTN the Jacobi eigenvalue algorithm is now complete!
332  // - The diagonals of diagMatrix are the eigenvalues
333  // - The transpose of transformationMatrix is the matrix whose columns are the eigenvectors
334  //
335  // Below we put this information into a reproducible form, i.e. we sort the eigenvalues by size, and normalize the eigenvectors
336 
337  // Extract the eigenvalues from the diagonalised matrix along with their indices
338  std::vector< std::pair<size_t, float> > eigenvalueIndexPairs;
339  for (size_t i = 0; i < size; ++i)
340  eigenvalueIndexPairs.emplace_back(i, diagMatrix.At(i, i));
341 
342  // Sort the eigenvalues by their magnitude (largest first)
343  std::sort(eigenvalueIndexPairs.begin(), eigenvalueIndexPairs.end(), [] (const auto &a, const auto &b) { return a.second > b.second; });
344 
345  // Get the elements of the matrix whose rows are the eigenvectors of the input matrix - sorted by eigenvalue and normalised to unity
346  std::vector<float> eigenvectorMatrixElements;
347  std::vector<float> eigenvalues;
348  for (const auto &[iRow, eigenvalue] : eigenvalueIndexPairs)
349  {
350  // ATTN we have a choice of the sign of the normalisation
351  // Here we choose the sign such that the largest element of the eigenvector is positive
352  float maxValue = -std::numeric_limits<float>::max();
353  int sign = 1;
354  float norm2 = 0.f;
355  for (size_t iCol = 0; iCol < size; ++iCol)
356  {
357  // Sum the squares of the elements of the eigenvector
358  const auto value = transformationMatrix.At(iRow, iCol);
359  norm2 += std::pow(value, 2.f);
360 
361  // Get the sign of the largest element of the eigenvector
362  if (std::abs(value) >= maxValue)
363  {
364  maxValue = std::abs(value);
365  sign = (value > 0.f) ? 1 : -1;
366  }
367  }
368 
369  if (norm2 <= std::numeric_limits<float>::epsilon())
370  throw std::logic_error("UBMatrixHelper::GetEigenDecomposition - Found an eigenvector with invalid norm");
371 
372  const auto norm = std::pow(norm2, 0.5f) * sign;
373 
374  // Normalise the elements of the eigenvector and store them
375  for (size_t iCol = 0; iCol < size; ++iCol)
376  eigenvectorMatrixElements.push_back(transformationMatrix.At(iRow, iCol) / norm);
377 
378  // Store the current eigenvalue
379  eigenvalues.push_back(eigenvalue);
380  }
381 
382  // Return the output matrices as a pair
383  return std::pair<UBMatrix, UBMatrix> (
384  { eigenvalues, size, 1 },
385  UBMatrix(eigenvectorMatrixElements, size, size).GetTranspose()
386  );
387 }

409 inline UBMatrix UBMatrixHelper::GetMatrixFromPartialEigenDecomposition(const UBMatrix &eigenvalues, const UBMatrix &eigenvectorMatrix)
410 {
411  // Check that the input matrices have consistent dimensions
412  if (eigenvalues.GetColumns() != 1u)
413  throw std::invalid_argument("UBMatrixHelper::GetMatrixFromPartialEigenDecomposition - input eigenvalues is not a column vector");
414 
415  const auto nEigenvalues = eigenvalues.GetRows();
416 
417  if (eigenvectorMatrix.GetColumns() != nEigenvalues)
418  throw std::invalid_argument("UBMatrixHelper::GetMatrixFromPartialEigenDecomposition - the number of columns of the input eigenvector matrix, doesn't match the number of supplied eigenvalues");
419 
420  const auto size = eigenvectorMatrix.GetRows();
421  if (size < nEigenvalues)
422  throw std::invalid_argument("UBMatrixHelper::GetMatrixFromPartialEigenDecomposition - the number of rows of the input eigenvector matrix is smaller than the number of columns");
423 
424  // Get the diagonal matrix with eigenvalues along the diagonal
425  const auto eigenvalueMatrix = ubsmear::UBMatrixHelper::GetDiagonalMatrix(eigenvalues);
426 
427  // Perform the transformation
428  // ATTN here we are assuming that the output matrix is real and symmetric, and hence the inverse of the matrix of eigenvectors is given
429  // by it's transpose. This is not true for a general matrix.
430  return (eigenvectorMatrix * eigenvalueMatrix * eigenvectorMatrix.GetTranspose());
431 }

435 inline std::pair<size_t, size_t> UBMatrixHelper::GetLargestOffDiagonalElement(const UBMatrix &matrix)
436 {
437  float maxElementValue = -std::numeric_limits<float>::max();
438  std::pair<size_t, size_t> maxElementIndex;
439  bool foundElement = false;
440 
441  for (size_t iRow = 0; iRow < matrix.GetRows(); ++iRow)
442  {
443  for (size_t iCol = 0; iCol < matrix.GetColumns(); ++iCol)
444  {
445  // Skip diagonal entries
446  if (iRow == iCol)
447  continue;
448 
449  // Skip any values that aren't bigger then the current largest value found
450  const auto value = std::abs(matrix.At(iRow, iCol));
451  if (value < maxElementValue)
452  continue;
453 
454  // Store this element
455  foundElement = true;
456  maxElementValue = value;
457  maxElementIndex.first = iRow;
458  maxElementIndex.second = iCol;
459  }
460  }
461 
462  if (!foundElement)
463  throw std::logic_error("UBMatrixHelper::GetLargestOffDiagonalElement - Largest off-diagonal element doesn't exist");
464 
465  return maxElementIndex;
466 }

498 inline bool UBMatrixHelper::IsSymmetric(const UBMatrix &matrix)
499 {
500  // A symmetric matrix is also square
501  if (!UBMatrixHelper::IsSquare(matrix))
502  return false;
503 
504  for (size_t iRow = 0; iRow < matrix.GetRows(); ++iRow)
505  {
506  for (size_t iCol = 0; iCol < iRow; ++iCol)
507  {
508  // Check if the matrix elements are identical under the exchange of rows and columns
509  const auto diff = std::abs(matrix.At(iRow, iCol) - matrix.At(iCol, iRow));
510  if (diff >= std::numeric_limits<float>::epsilon())
511  return false;
512  }
513  }
514 
515  return true;
516 }
