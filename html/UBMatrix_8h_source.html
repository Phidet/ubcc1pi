294 inline UBMatrix UBMatrix::GetRowNormalized() const
295 {
296  std::vector<float> elements;
297 
298  for (size_t iRow = 0; iRow < m_nRows; ++iRow)
299  {
300  // Get the sum of the elements in this row
301  float rowSum = 0.f;
302  for (size_t iCol = 0; iCol < m_nCols; ++iCol)
303  {
304  rowSum += this->At(iRow, iCol);
305  }
306 
307  if (std::abs(rowSum) <= std::numeric_limits<float>::epsilon())
308  throw std::logic_error("UBMatrix::GetRowNormalized - Found a row which sums to zero, can't normalize it");
309 
310  // Normalize the elements in the row
311  const auto norm = 1.f / rowSum;
312  for (size_t iCol = 0; iCol < m_nCols; ++iCol)
313  {
314  elements.push_back(this->At(iRow, iCol) * norm);
315  }
316  }
317 
318  return UBMatrix(elements, m_nRows, m_nCols);

350 {
