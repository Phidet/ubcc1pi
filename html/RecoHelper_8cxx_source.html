323 float RecoHelper::GetBraggLikelihood(const art::Ptr<anab::ParticleID> &pid, const int &pdg, const anab::kTrackDir &dir, const float yzAngle, const float sin2AngleThreshold, const unsigned int nHitsU, const unsigned int nHitsV)
324 {
325  const auto piBy3 = std::acos(0.5f);
326 
327  const auto likelihoodW = RecoHelper::GetBraggLikelihood(pid, pdg, geo::kW, dir);
328  const bool isTrackAlongWWire = (std::pow(std::sin(yzAngle), 2) < sin2AngleThreshold);
329  const auto hasW = (likelihoodW > -1.f && !isTrackAlongWWire);
330 
331  if (hasW)
332  return likelihoodW;
333 
334  // Otherwise the track is along the W wire direction, so just average the other two planes weighted by the number of degrees of freedom
335  const auto likelihoodU = RecoHelper::GetBraggLikelihood(pid, pdg, geo::kU, dir);
336  const bool isTrackAlongUWire = (std::pow(std::sin(yzAngle - piBy3), 2) < sin2AngleThreshold);
337  const auto hasU = (likelihoodU > -1.f && !isTrackAlongUWire);
338 
339  const auto likelihoodV = RecoHelper::GetBraggLikelihood(pid, pdg, geo::kV, dir);
340  const bool isTrackAlongVWire = (std::pow(std::sin(yzAngle + piBy3), 2) < sin2AngleThreshold);
341  const auto hasV = (likelihoodV > -1.f && !isTrackAlongVWire);
342 
343  const auto dofU = hasU ? static_cast<float>(nHitsU) : 0.f;
344  const auto dofV = hasV ? static_cast<float>(nHitsV) : 0.f;
345  const auto dofUV = dofU + dofV;
346 
347  const auto weightU = hasU ? (likelihoodU * dofU) : 0.f;
348  const auto weightV = hasV ? (likelihoodV * dofV) : 0.f;
349 
350  if ((!hasU && !hasV) || (nHitsU == 0 && nHitsV == 0) || dofUV <= std::numeric_limits<float>::epsilon())
351  return -std::numeric_limits<float>::max();
352 
353  return (weightU + weightV) / dofUV;
354 }

416 float RecoHelper::GetTrackWiggliness(const art::Ptr<recob::Track> &track)
417 {
418  const auto validPoints = RecoHelper::GetValidPoints(track);
419  if (validPoints.size() < 3)
420  return 0.f;
421 
422  std::vector<float> thetaVector;
423  float thetaSum = 0.f;
424  for (unsigned int i = 1; i < validPoints.size(); ++i)
425  {
426  const auto dir = track->DirectionAtPoint(validPoints.at(i));
427  const auto dirPrev = track->DirectionAtPoint(validPoints.at(i - 1));
428 
429  // Bind between -1 and 1 at floating precision to avoid issues with cast from double
430  const auto cosTheta = std::min(1.f, std::max(-1.f, static_cast<float>(dir.Dot(dirPrev))));
431  const auto theta = std::acos(cosTheta);
432 
433  thetaSum += theta;
434  thetaVector.push_back(theta);
435  }
436 
437  const auto thetaMean = thetaSum / static_cast<float>(thetaVector.size());
438 
439  float thetaDiffSum = 0.f;
440  for (const auto &theta : thetaVector)
441  {
442  thetaDiffSum += std::pow(theta - thetaMean, 2);
443  }
444 
445  const auto variance = thetaDiffSum / static_cast<float>(thetaVector.size() - 1);
446 
447  return std::sqrt(variance);
448 }

471 float RecoHelper::GetTruncatedMeandEdxAtTrackStart(const std::vector<float> dedxPerHit, const std::vector<float> &residualRangePerHit, const unsigned int nHitsToSkip, const float lengthFraction)
472 {
473  if (dedxPerHit.size() != residualRangePerHit.size())
474  throw cet::exception("RecoHelper::GetTruncatedMeandEdxAtTrackStart") << " - dEdx per hit and residual range vectors have different sizes" << std::endl;
475 
476  // Check if the variable is calculable
477  if (residualRangePerHit.size() <= nHitsToSkip)
478  return -std::numeric_limits<float>::max();
479 
480  // Make make the vector of pairs to keep track of the incides, and find the maximum residual range
481  std::vector<std::pair<float, unsigned int> > residualRangeIndices;
482  float maxResidualRange = -std::numeric_limits<float>::max();
483  for (unsigned int i = 0; i < residualRangePerHit.size(); ++i)
484  {
485  const auto residualRange = residualRangePerHit.at(i);
486  maxResidualRange = std::max(maxResidualRange, residualRange);
487 
488  residualRangeIndices.emplace_back(residualRange, i);
489  }
490 
491  const auto residualRangeCutoff = maxResidualRange * lengthFraction;
492 
493  // Sort the residual ranges such that the largest residual range (closest to the start of the track) is first
494  std::sort(residualRangeIndices.begin(), residualRangeIndices.end(), [](auto &a, auto &b) {
495  return a.first > b.first;
496  });
497 
498  // Get the dEdx of the hits at the start of the track
499  std::vector<float> dedxPerHitAtStart;
500  for (unsigned int i = nHitsToSkip; i < residualRangeIndices.size(); ++i)
501  {
502  const auto entry = residualRangeIndices.at(i);
503  const auto residualRange = entry.first;
504  const auto hitIndex = entry.second;
505 
506  // ATTN small residual ranges are at the start of the track
507  if (residualRange < residualRangeCutoff)
508  continue;
509 
510  dedxPerHitAtStart.push_back(dedxPerHit.at(hitIndex));
511  }
512 
513  const auto nHits = dedxPerHitAtStart.size();
514  if (nHits == 0)
515  return -std::numeric_limits<float>::max();
516 
517  // Sort the dEdx so we can find the median
518  std::sort(dedxPerHitAtStart.begin(), dedxPerHitAtStart.end());
519  const auto median = dedxPerHitAtStart.at(nHits / 2);
520 
521  // Now find the mean
522  float total = 0.f;
523  for (const auto &dEdx : dedxPerHitAtStart)
524  total += dEdx;
525 
526  const auto mean = total / static_cast<float>(nHits);
527 
528  // Now find the variance
529  float squareSum = 0.f;
530  for (const auto &dEdx : dedxPerHitAtStart)
531  squareSum += std::pow(dEdx - mean, 2);
532 
533  const auto variance = squareSum / static_cast<float>(nHits);
534 
535  // Get the mean dEdx of the hits within one standard deviation of the median
536  float truncatedTotal = 0.f;
537  unsigned int nTruncatedHits = 0;
538  for (const auto &dEdx : dedxPerHitAtStart)
539  {
540  if (std::pow(dEdx - median, 2) > variance)
541  continue;
542 
543  truncatedTotal += dEdx;
544  nTruncatedHits++;
545  }
546 
547  if (nTruncatedHits == 0)
548  return -std::numeric_limits<float>::max();
549 
550  const auto truncatedMean = truncatedTotal / static_cast<float>(nTruncatedHits);
551 
552  return truncatedMean;
553 }
