470 void EventFactory::PopulateEventRecoParticleMCSFitInfo(const art::Ptr<recob::MCSFitResult> &mcsFitResult, Event::Reco::Particle &particle)
471 {
472  // ATTN the MCS fitter will default to the maximum sampled momentum if the likelihood scan doesn't converge
473  // Unfortunately, the fitter doesn't provide a flag to identify the cases that didn't converge
474  // With the current configuration, this maximum sample is around 15 GeV, so as a workaround here we apply a threshold below this value
475  // to avoid saving poor fit results (i.e. we are classing this as a fit failure). If the MCS fitter wasn't able to run at all (maybe
476  // there were insufficient hits) then it will return a negative momentum - so we check for this too!
477  // // We do however still save the likelihood in-case that is useful
478  const auto maxMomentum = 14.f;
479 
480 
481  // Forward fit
482  const auto forwardMomentum = mcsFitResult->fwdMomentum();
483  if (forwardMomentum >= 0.f && forwardMomentum < maxMomentum)
484  {
485  particle.mcsMomentumForwardMuon.Set(mcsFitResult->fwdMomentum());
486  particle.mcsMomentumUncertaintyForwardMuon.Set(mcsFitResult->fwdMomUncertainty());
487  }
488 
489  // Backward fit
490  const auto backwardMomentum = mcsFitResult->bwdMomentum();
491  if (backwardMomentum >= 0.f && backwardMomentum < maxMomentum)
492  {

576 void EventFactory::PopulateEventRecoParticleCalorimetryInfo(const Config &config, const CalorimetryVector &calos, Event::Reco::Particle &particle)
577 {
578  float uWeight = 0.f;
579  float vWeight = 0.f;
580  for (const auto &calo : calos)
581  {
582  const auto dedxPerHit = calo->dEdx();
583  const auto residualRangePerHit = calo->ResidualRange();
584  const auto truncatedMeandEdx = RecoHelper::GetTruncatedMeandEdxAtTrackStart(dedxPerHit, residualRangePerHit, config.NHitsToSkip(), config.LengthFraction());
585 
586  Member<float> *pMember = nullptr;
587 
588  switch (calo->PlaneID().Plane)
589  {
590  case geo::kU:
591  pMember = &particle.truncatedMeandEdxU;
592  uWeight = std::max(0.f, (dedxPerHit.size() * config.LengthFraction()) - config.NHitsToSkip());
593  break;
594  case geo::kV:
595  pMember = &particle.truncatedMeandEdxV;
596  vWeight = std::max(0.f, (dedxPerHit.size() * config.LengthFraction()) - config.NHitsToSkip());
597  break;
598  case geo::kW:
599  pMember = &particle.truncatedMeandEdxW;
600  break;
601  default: break;
602  }
603 
604  if (pMember && truncatedMeandEdx > -1.f)
605  pMember->Set(truncatedMeandEdx);
606  }
607 
608  const auto yzAngle = particle.yzAngle();
609  const bool isTrackAlongWWire = (std::pow(std::sin(yzAngle), 2) < config.Sin2AngleThreshold());
610  if (!isTrackAlongWWire && particle.truncatedMeandEdxW.IsSet())
611  {
612  particle.truncatedMeandEdx.Set(particle.truncatedMeandEdxW());
613  }
614  else
615  {
616  const auto hasU = particle.truncatedMeandEdxU.IsSet() && uWeight > 0.f;
617  const auto hasV = particle.truncatedMeandEdxV.IsSet() && vWeight > 0.f;
618 
619  if (hasU || hasV)
620  {
621  float truncatedMeandEdx = 0.f;
622  if (hasU) truncatedMeandEdx += particle.truncatedMeandEdxU() * uWeight;
623  if (hasV) truncatedMeandEdx += particle.truncatedMeandEdxV() * vWeight;
624 
625  truncatedMeandEdx /= (uWeight + vWeight);
626  particle.truncatedMeandEdx.Set(truncatedMeandEdx);
627  }
628  }
629 }

633 void EventFactory::PopulateEventRecoSliceInfo(const art::Event &event, const Config &config, Event::Reco &reco)
634 {
635  // Reco slice information
636  const auto slices = CollectionHelper::GetCollection<recob::Slice>(event, config.SliceLabel());
637  const auto slicesToHits = CollectionHelper::GetAssociation<recob::Slice, recob::Hit>(event, config.SliceLabel());
638  const auto pfParticlesToSlices = CollectionHelper::GetAssociation<recob::PFParticle, recob::Slice>(event, config.PFParticleLabel(), config.SliceLabel());
639  const auto slicesToPFParticles = CollectionHelper::GetReversedAssociation(pfParticlesToSlices);
640  const auto pfParticlesToMetadata = CollectionHelper::GetAssociation<recob::PFParticle, larpandoraobj::PFParticleMetadata>(event, config.PFParticleLabel());
641 
642  // Work out which slice (if any) was selected as the neutrino
643  art::Ptr<recob::Slice> selectedSlice;
644  std::vector<bool> sliceIsSelectedAsNu;
645  for (const auto &slice : slices)
646  {
647  const auto isSelectedAsNu = RecoHelper::IsSliceSelectedAsNu(slice, slicesToPFParticles);
648  sliceIsSelectedAsNu.push_back(isSelectedAsNu);
649 
650  if (isSelectedAsNu)
651  {
652  if (selectedSlice.isNonnull())
653  throw cet::exception("AnalysisHelper::PopulateEventRecoSliceInfo") << " - Multiple slices selected as neutrino." << std::endl;
654 
655  selectedSlice = slice;
656  }
657  }
658 
659  const auto hasSelectedSlice = selectedSlice.isNonnull();
660 
661  // Get the topological scores of the slices
662  std::vector<float> sliceTopologicalScores;
663  for (const auto &slice : slices)
664  {
665  // Define the dummy value for slices without a topological score
666  float topologicalScore = -1.f;
667 
668  // The score of the slice is held in the metadata of the primary PFParticles in that slice
669  for (const auto &pfParticle : CollectionHelper::GetManyAssociated(slice, slicesToPFParticles))
670  {
671  if (pfParticle->IsPrimary())
672  {
673  const auto metadata = CollectionHelper::GetSingleAssociated(pfParticle, pfParticlesToMetadata);
674 
675  try
676  {
677  topologicalScore = RecoHelper::GetTopologicalScore(metadata);
678  break;
679  }
680  catch (const cet::exception &) {}
681  }
682  }
683 
684  // ATTN the output topological scores can contain dummy values - this keeps the vectors in sync.
685  sliceTopologicalScores.push_back(topologicalScore);
686 
687  if (slice == selectedSlice)
688  reco.selectedTopologicalScore.Set(topologicalScore);
689  }
690 
691  reco.nSlices.Set(slices.size());
692  reco.hasSelectedSlice.Set(hasSelectedSlice);
693  reco.sliceTopologicalScores.Set(sliceTopologicalScores);
694  reco.sliceIsSelectedAsNu.Set(sliceIsSelectedAsNu);
695 }

699 void EventFactory::PopulateEventTruthSliceInfo(const art::Event &event, const Config &config, Event::Truth &truth)
700 {
701  const auto slices = CollectionHelper::GetCollection<recob::Slice>(event, config.SliceLabel());
702  const auto slicesToHits = CollectionHelper::GetAssociation<recob::Slice, recob::Hit>(event, config.SliceLabel());
703  const auto pfParticlesToSlices = CollectionHelper::GetAssociation<recob::PFParticle, recob::Slice>(event, config.PFParticleLabel(), config.SliceLabel());
704  const auto slicesToPFParticles = CollectionHelper::GetReversedAssociation(pfParticlesToSlices);

706  const auto nuMCParticles = TruthHelper::GetMCParticlesFromMCTruth(event, config.MCTruthLabel(), config.MCParticleLabel(), nuMCTruth);
707  const auto hitsToIsNuInduced = BacktrackHelper::GetHitsToIsNuInducedMap(event, config.HitLabel(), config.MCParticleLabel(), config.BacktrackerLabel(), nuMCParticles);
708 
709  // Get the required mapping from slice to "is selected as neutrino"
710  SlicesToBool sliceToIsSelectedAsNu;
711  for (const auto &slice : slices)
712  {
713  const auto isSelectedAsNu = RecoHelper::IsSliceSelectedAsNu(slice, slicesToPFParticles);
714 
715  if (!sliceToIsSelectedAsNu.emplace(slice, isSelectedAsNu).second)
716  throw cet::exception("AnalysisHelper::PopulateEventRecoSliceInfo") << " - Repeated slices." << std::endl;
717  }
718 
719  const auto sliceMetadata = BacktrackHelper::SliceMetadata(slices, sliceToIsSelectedAsNu, slicesToHits, hitsToIsNuInduced);
720  const auto nNeutrinoHits = sliceMetadata.GetTotalNumberOfNuInducedHits();
721 
722  // Get the purities and completenesses of the slices
723  std::vector<float> slicePurities, sliceCompletenesses;
724  for (const auto &slice : slices)
725  {
726  const auto nHitsInSlice = sliceMetadata.GetNumberOfHits(slice);
727 
728  slicePurities.push_back(nHitsInSlice == 0 ? -std::numeric_limits<float>::max() : sliceMetadata.GetPurity(slice));
729  sliceCompletenesses.push_back(nNeutrinoHits == 0 ? -std::numeric_limits<float>::max() : sliceMetadata.GetCompleteness(slice));
730  }
731 
732  truth.slicePurities.Set(slicePurities);
733  truth.sliceCompletenesses.Set(sliceCompletenesses);
734 }
