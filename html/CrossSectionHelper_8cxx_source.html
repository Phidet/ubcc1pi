299 ubsmear::UBMatrix CrossSectionHelper::CrossSection::GetCrossSection(const ubsmear::UBMatrix &selected, const ubsmear::UBMatrix &backgrounds, const float integratedFlux, const float exposurePOT, const float nTargets) const
300 {
301  // Check the input event rate has the right dimensions
302  if (selected.GetColumns() != 1)
303  throw std::invalid_argument("CrossSection::GetCrossSection - Input event rate isn't a column vector");
304 
305  if (selected.GetRows() != m_metadata.GetNBins())
306  throw std::invalid_argument("CrossSection::GetCrossSection - Input event rate has the wrong number of bins");
307 
308  // Check the input backgrounds have the right dimensions
309  if (backgrounds.GetColumns() != 1)
310  throw std::invalid_argument("CrossSection::GetCrossSection - Input background rate isn't a column vector");
311 
312  if (backgrounds.GetRows() != m_metadata.GetNBins())
313  throw std::invalid_argument("CrossSection::GetCrossSection - Input background rate has the wrong number of bins");
314 
315  // Get the normalisation factor
316  const auto norm = integratedFlux * exposurePOT * nTargets;
317  if (norm <= std::numeric_limits<float>::epsilon())
318  throw std::invalid_argument("CrossSection::GetCrossSection - Product of flux, exposure and targets non-positive");
319 
320  // Get the column vector of bin widths
321  const auto binWidths = this->GetBinWidths();
322 
323  // Now get the flux-integrated, forward-folded cross-section by
324  // - Subtracting the backgrounds
325  // - Scaling by the normalisation factor
326  // - Scaling (element wise) by the bin-widths
327  return ubsmear::ElementWiseOperation(
328  (selected - backgrounds),
329  binWidths * norm,
330  [](const float numerator, const float denominator) {
331 
332  if (denominator <= std::numeric_limits<float>::epsilon())
333  throw std::logic_error("CrossSection::GetCrossSection - Found a bin in which the denominator (flux * POT * nTargets * binWidth) is <= 0");
334 
335  return numerator / denominator;
336  }
337  );
338 }

343 {

345  const auto allTrue = CrossSectionHelper::GetMatrixFromHist(pSignal_true);
346  const auto selectedRecoTrue = CrossSectionHelper::GetMatrixFromHist(pSignal_selected_recoTrue);
347 
348  // Check theh matrices have the expected sizes
349  if (!ubsmear::UBMatrixHelper::IsSquare(selectedRecoTrue))
350  throw std::invalid_argument("CrossSection::GetSmearingMatrix - input reco-true matrix isn't square");

357  std::vector<float> elements;
358  for (unsigned int iReco = 0; iReco < nBins; ++iReco)
359  {
360  for (unsigned int iTrue = 0; iTrue < nBins; ++iTrue)

362  // Get the total number of signal events in this true bin

707 CrossSectionHelper::SystBiasCovariancePair CrossSectionHelper::CrossSection::GetDistributionParams(const std::function<std::shared_ptr<ubsmear::UBMatrix>(const unsigned int)> &func, const unsigned int nUniverses, const std::shared_ptr<ubsmear::UBMatrix> &pNominal) const
708 {
709  if (nUniverses == 0)
710  throw std::logic_error("CrossSection::GetDisributionParams - No universes supplied");
711 
712  // Insist the nominal input is a column vector
713  if (pNominal->GetColumns() != 1)
714  throw std::logic_error("CrossSection::GetDisributionParams - Input nominal is not a column vector");
715 
716  // Get the number of bins
717  const auto nBins = pNominal->GetRows();
718 
719  // Setup some empty matrices to hold the mean vector and error matrices
720  auto meanSum = ubsmear::UBMatrixHelper::GetZeroMatrix(nBins, 1);
721  auto errorMatrixSum = ubsmear::UBMatrixHelper::GetZeroMatrix(nBins, nBins);
722 
723  // Loop over the universes
724  unsigned int nValidUniverses = 0u;
725  for (unsigned int iUni = 0; iUni < nUniverses; ++iUni)
726  {
727  //// BEGIN DEBUG
728  AnalysisHelper::PrintLoadingBar(iUni, nUniverses);
729  //// END DEBUG
730 
731  // Get the value in this universe
732  const auto pUniverse = func(iUni);
733 
734  // Insist that we get a valid quanitiy in this universes
735  if (!pUniverse)
736  continue;
737 
738  // Count the number of valid universes
739  nValidUniverses++;
740 
741  // Loop over the bins
742  // ATTN for speed we don't expcitly check here if the universe has the same number of bins as the nominal. Instead we rely on the
743  // internal range checking of the ubsmear::UBMatrix class
744  for (unsigned int iBin = 0; iBin < nBins; ++iBin)
745  {
746  // Add up the universes
747  meanSum.SetElement(iBin, 0, meanSum.At(iBin, 0) + pUniverse->At(iBin, 0));
748 
749  // Loop over the bins again
750  // ATTN the error matrix is symmetic so here we only set one half of the off-diagonals within the universe loop and then copy
751  // them over to the other half afterwards. This is done for performance reasons as it halves the number of inserts required
752  const auto diffI = pUniverse->At(iBin, 0) - pNominal->At(iBin, 0);
753  for (unsigned int jBin = 0; jBin <= iBin; ++jBin)
754  {
755  const auto diffJ = pUniverse->At(jBin, 0) - pNominal->At(jBin, 0);
756 
757  // Add up the error matrix elements
758  errorMatrixSum.SetElement(iBin, jBin, errorMatrixSum.At(iBin, jBin) + diffI*diffJ);
759  }
760  }
761  }
762 
763  std::cout << "DEBUG - Valid universes: " << nValidUniverses << " / " << nUniverses << std::endl;
764 
765  // Scale the sums by the number of universes
766  if (nValidUniverses == 0)
767  throw std::logic_error("CrossSection::GetDisributionParams - Desired quantity was invalid in all universes");
768 
769  const auto scaleFactor = 1.f / static_cast<float>(nValidUniverses);
770  const auto mean = meanSum * scaleFactor;
771  const auto errorMatrix = errorMatrixSum * scaleFactor;
772 
773  // Get the bias vector
774  const auto pBias = std::make_shared<ubsmear::UBMatrix>(mean - *pNominal);
775 
776  // Get the covariance matrix
777  auto pCovarianceMatrix = std::make_shared<ubsmear::UBMatrix>(ubsmear::UBMatrixHelper::GetZeroMatrix(nBins, nBins));
778  for (unsigned int iBin = 0; iBin < nBins; ++iBin)
779  {
780  for (unsigned int jBin = 0; jBin <= iBin; ++jBin)
781  {
782  // ATTN here we use the fact that V_ij = E_ij - b_i*b_j
783  // - E_ij = error matrix element
784  // - V_ij = covariance matrix element
785  // - b_i = bias vector element
786  //
787  // This is done instead of calculating the covariance directly as this only requires one pass through the universe loop.
788  const auto covariance = errorMatrix.At(iBin, jBin) - pBias->At(iBin, 0)*pBias->At(jBin, 0);
789 
790  pCovarianceMatrix->SetElement(iBin, jBin, covariance);
791  pCovarianceMatrix->SetElement(jBin, iBin, covariance);
792  }
793  }
794 
795  return { pBias, pCovarianceMatrix };
796 }

800 CrossSectionHelper::SystBiasCovariancePair CrossSectionHelper::CrossSection::GetDistributionParamsUnisim(const std::shared_ptr<ubsmear::UBMatrix> &pVaried, const std::shared_ptr<ubsmear::UBMatrix> &pCentralValue, const std::shared_ptr<ubsmear::UBMatrix> &pNominal) const
801 {
802  // Check the input matrices are valid
803  if (!pVaried || !pCentralValue || !pNominal)
804  throw std::invalid_argument("CrossSection::GetDistributionParamsUnisim - One or more of the inputs are not valid");
805 
806  // Check ths input matrices have the desired dimensions
807  if (pVaried->GetColumns() != 1 || pCentralValue->GetColumns() != 1 || pNominal->GetColumns() != 1)
808  throw std::invalid_argument("CrossSection::GetDistributionParamsUnisim - One or more of the inputs are not a column vector");
809 
810  const auto nBins = pVaried->GetRows();
811  if (pCentralValue->GetRows() != nBins || pNominal->GetRows() != nBins)
812  throw std::invalid_argument("CrossSection::GetDistributionParamsUnisim - Input column vectors have different sizes");
813 
814  // Get the fractional bias of the cross-section away from the central value
815  const auto fracBias = ubsmear::ElementWiseOperation(
816  (*pVaried - *pCentralValue), *pCentralValue,
817  [](const float numerator, const float denominator)
818  {
819  // ATTN here we decide what to do if there's a bin of the central-value that's zero
820  // In this case we don't have any information, so here we set the fractional bias to 0.
821  if (std::abs(denominator) <= std::numeric_limits<float>::epsilon())
822  return 0.f;
823 
824  return numerator / denominator;
825  }
826  );
827 
828  // Apply the fractional bias to the nominal cross-section
829  const auto bias = ubsmear::ElementWiseOperation(
830  fracBias, *pNominal,
831  [](const float lhs, const float rhs) { return lhs * rhs; }
832  );
833 
834  // Return the result (use a zero matrix for the covariance)
835  return {
836  std::make_shared<ubsmear::UBMatrix>(bias),
837  std::make_shared<ubsmear::UBMatrix>(ubsmear::UBMatrixHelper::GetZeroMatrix(nBins, nBins))
838  };
839 }
