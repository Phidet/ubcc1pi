44

48  {
49  AnalysisHelper::PrintLoadingBar(i, nEvents);
50 
51  reader.LoadEvent(i);
52 
53  // Get the nominal event weight, scaled by the sample normalisation
54  const auto weight = AnalysisHelper::GetNominalEventWeight(pEvent) * normalisation;
55 
56  // For overlay & dirt samples only, count all events before any selection cuts
57  // ATTN The EXT and BNB data has been filtered through the CC inclusive selection, so we can't get the count before any cuts were applied
58  if (sampleType == AnalysisHelper::Overlay || sampleType == AnalysisHelper::Dirt)
59  {
60  counter.CountEvent("all", sampleType, pEvent, weight);
61  }
62

65 
66  // Check which of the cuts the event passed
67  for (const auto &cutName : selection.GetCuts())
68  {
69  // If the event didn't pass this cut, then move on
70  if (!SelectionHelper::IsCutPassed(cutsPassed, cutName))
71  break;
72 
73  // Count the event, and use the cut name as the "tag"
74  counter.CountEvent(cutName, sampleType, pEvent, weight);
75  }
76  }
77  }
78

92  }
93
