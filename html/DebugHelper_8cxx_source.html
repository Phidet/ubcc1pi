40 void DebugHelper::PrintSummary(const MCParticleVector &particles, const unsigned int depth)
41 {
42  // Count the number of particles with each pdg code
43  std::unordered_map<int, MCParticleVector> pdgToMCParticleMap;
44  std::vector<int> usedPdgs;
45 
46  for (const auto &particle : particles)
47  {
48  const auto pdg = particle->PdgCode();
49  pdgToMCParticleMap[pdg].push_back(particle);
50 
51  if (std::find(usedPdgs.begin(), usedPdgs.end(), pdg) == usedPdgs.end())
52  usedPdgs.push_back(pdg);
53  }
54 
55  // Print the summary
56  std::sort(usedPdgs.begin(), usedPdgs.end());
57  for (const auto &pdg : usedPdgs)
58  {
59  DebugHelper::PrintProperty("# (" + std::to_string(pdg) + ")", pdgToMCParticleMap.at(pdg).size(), depth);
60  }
61 }

66 {
67  // Convert the interaction type to a human readable string

77  break;
78  case 16:
79  nuType = brief ? "nutau" : "Tau Neutrino";
80  break;
81  case -12:
