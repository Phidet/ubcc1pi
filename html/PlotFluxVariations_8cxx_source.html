24 void PlotFluxVariations(const Config &config)
25 {
26  // Setup the dimensions
27  CrossSectionHelper::SystDimensionsMap fluxDimensions;
28  for (const auto &[paramName, nUniverses] : config.extractXSecs.fluxDimensions)
29  {
30  // If this is a parameter that's stored directly in the event weights, then just use it
31  if (config.extractXSecs.mutuallyExclusiveDimensions.count(paramName) == 0)
32  {
33  fluxDimensions.emplace(paramName, nUniverses);
34  continue;
35  }
36 
37  // Otherwise, we have a parameter built from a set of mutually exclusive parameters
38  // For the sake of these validation plots, we will use each parameter individually
39  const auto &[mutuallyExclusiveParams, nUniversesCheck] = config.extractXSecs.mutuallyExclusiveDimensions.at(paramName);
40  if (nUniverses != nUniversesCheck)
41  throw std::invalid_argument("PlotFluxVariations - Wrong number of universes for mutually exclusive param: " + paramName);
42 
43  for (const auto &mutuallyExclusiveParamName : mutuallyExclusiveParams)
44  fluxDimensions.emplace(mutuallyExclusiveParamName, nUniverses);
45  }
46 
47  // Open the flux file for reading
48  // Get the scaling factor to go from the event rate in the flux file, to the flux itself
49  // Here we get the flux in neutrinos/POT/bin/cm2 by scaling the event rate (in the samples) down by POT and the cross-sectional area of the active volume
50  // Here we also scale up the fluxes by 1e10 for the sake of comparison just so we are working with reasonable numbers
51  const float unitsScaling = 1e10;
52  const auto fluxScaleFactor = unitsScaling / (config.flux.pot * (GeometryHelper::highX - GeometryHelper::lowX) * (GeometryHelper::highY - GeometryHelper::lowY));
53 
54  // Get the flux bin edges and content
55  const auto fluxHistNames = CrossSectionHelper::GetNominalFluxHistNames(config.flux.nuPdgsSignal, config.flux.nuPdgToHistName, config.flux.nomHistPattern);
56  const auto &[fluxBinEdges, fluxBinValuesNominal] = CrossSectionHelper::ReadNominalFlux(config.flux.fileName, fluxHistNames, config.flux.pot);
57 
58  // Setup the flux reweightor
59  CrossSectionHelper::FluxReweightor fluxReweightor(fluxBinEdges, fluxBinValuesNominal, fluxDimensions);
60  std::cout << "Integrated nominal flux: " << fluxReweightor.GetIntegratedNominalFlux() << std::endl;
61 
62  std::cout << "Flux includes:" << std::endl;
63  for (const auto &fluxHistName : fluxHistNames)
64  {
65  std::cout << " - " << fluxHistName << std::endl;
66  }
67 
68  // Open the flux file
69  TFile *pFluxFile = new TFile(config.flux.fileName.c_str(), "READ");
70  if (!pFluxFile->IsOpen())
71  throw std::invalid_argument("PlotFluxVariations - Can't open flux file: " + config.flux.fileName);
72 
73  // Open the input file for reading and enable the branches with systematic event weights
74  FileReader reader(config.files.overlaysFileName);
75  reader.EnableSystematicBranches();
76 
77  // Get the bound event address
78  auto pEvent = reader.GetBoundEventAddress();
79 
80  // Loop over the events
81  const auto nEvents = reader.GetNumberOfEvents();
82  for (unsigned int i = 0; i < nEvents; ++i)
83  {
84  AnalysisHelper::PrintLoadingBar(i, nEvents);
85  reader.LoadEvent(i);
86 
87  // Only use the desired neutrino flavours
88  if (std::find(config.flux.nuPdgsSignal.begin(), config.flux.nuPdgsSignal.end(), pEvent->truth.nuPdgCode()) == config.flux.nuPdgsSignal.end())
89  continue;
90 
91  // Skip events that aren't in the active volume
92  if (!AnalysisHelper::IsInActiveVolume(pEvent->truth.nuVertex()))
93  continue;
94 
95  // Get the nominal event weight
96  const auto weight = AnalysisHelper::GetNominalEventWeight(pEvent);
97 
98  // Get the flux weights
99  const auto fluxWeights = CrossSectionHelper::GetWeightsMap(pEvent->truth, fluxDimensions);
100 
101  // Add this event to the flux reweightor
102  fluxReweightor.AddEvent(pEvent->truth.nuEnergy(), weight, fluxWeights);
103  }
104 
105  // Setup a table for the RMS difference between the fluxes in the files and using the reweighting method
106  FormattingHelper::Table table({"Parameter", "True flux STD", "", "Frac diff RMS"});
107 
108  // Get the integrated flux in each universe
109  for (const auto &[paramName, nUniverses] : fluxDimensions)
110  {
111  std::cout << "Parameter name: " << paramName << std::endl;
112 
113  // Get the directories correponding to this parameter
114  std::unordered_map<int, TDirectoryFile *> nuPdgToDirMap;
115  for (const auto &nuPdg : config.flux.nuPdgsSignal)
116  {
117  const auto nuNameIter = config.flux.nuPdgToHistName.find(nuPdg);
118  if (nuNameIter == config.flux.nuPdgToHistName.end())
119  throw std::logic_error("PlotFluxVariations - Can't find name for PDG code: " + std::to_string(nuPdg));
120 
121  const auto &nuName = nuNameIter->second;
122  const auto dirName = std::regex_replace(std::regex_replace(config.flux.variationDirPattern, std::regex("NEUTRINO"), nuName), std::regex("PARAMNAME"), paramName);
123  const auto pDir = static_cast<TDirectoryFile *>(pFluxFile->Get(dirName.c_str()));
124  if (!pDir)
125  throw std::logic_error("PlotFluxVariations - Can't get TDirectoryFile: " + dirName);
126 
127  nuPdgToDirMap.emplace(nuPdg, pDir);
128  }
129 
130  // Store the true fluxes, and the fractional difference when using reweighting
131  std::vector<float> trueFluxes, fractionalDiffs;
132 
133  // Keep track of the min and max fluxes seen
134  float minFlux = +std::numeric_limits<float>::max();
135  float maxFlux = -std::numeric_limits<float>::max();
136  float maxFracDiff = -std::numeric_limits<float>::max();
137 
138  // Loop over the universes
139  for (unsigned int iUni = 0; iUni < nUniverses; ++iUni)
140  {
141  // Get the reweighted flux
142  const auto fluxReweighted = fluxReweightor.GetIntegratedFluxVariation(paramName, iUni);
143 
144  // Get the flux from the file for this universe
145  float fluxUni = 0.f;
146  for (const auto &nuPdg : config.flux.nuPdgsSignal)
147  {
148  // Get the name of the flux histogram
149  const auto nuName = config.flux.nuPdgToHistName.at(nuPdg);
150  const auto fluxName = std::regex_replace(std::regex_replace(std::regex_replace(config.flux.variationHistPattern, std::regex("PARAMNAME"), paramName), std::regex("UNIVERSEINDEX"), std::to_string(iUni)), std::regex("NEUTRINO"), nuName);
151  const auto pFluxUni = static_cast<TH1F *>(nuPdgToDirMap.at(nuPdg)->Get(fluxName.c_str()));
152  if (!pFluxUni)
153  throw std::logic_error("PlotFluxVariations - Can't get TH1F: " + fluxName);
154 
155  fluxUni += pFluxUni->Integral() * fluxScaleFactor;
156  }
157 
158  float fracDiff = 0.f;
159  if (std::abs(fluxUni) > std::numeric_limits<float>::epsilon())
160  {
161  fracDiff = (fluxReweighted - fluxUni) / fluxUni;
162  }
163 
164  minFlux = std::min(minFlux, fluxUni);
165  maxFlux = std::max(maxFlux, fluxUni);
166  maxFracDiff = std::max(maxFracDiff, std::abs(fracDiff));
167 
168  // Store the result
169  trueFluxes.push_back(fluxUni);
170  fractionalDiffs.push_back(fracDiff);
171  }
172 
173  // Make a histogram for the frational differences
174  auto pHistTrueFlux = std::make_shared<TH1F>(("trueFlux_" + paramName).c_str(), "", 50, minFlux, maxFlux);
175  auto pHistFracDiff = std::make_shared<TH1F>(("fracDiff_" + paramName).c_str(), "", 50, -maxFracDiff, maxFracDiff);
176  auto pHist2D = std::make_shared<TH2F>(("fracDiff-vs-trueFlux_" + paramName).c_str(), "", 50, minFlux, maxFlux, 50, -maxFracDiff, maxFracDiff);
177  for (unsigned int iUni = 0; iUni < nUniverses; ++iUni)
178  {
179  pHistTrueFlux->Fill(trueFluxes.at(iUni));
180  pHistFracDiff->Fill(fractionalDiffs.at(iUni));
181  pHist2D->Fill(trueFluxes.at(iUni), fractionalDiffs.at(iUni));
182  }
183 
184  auto pCanvas = PlottingHelper::GetCanvas();
185  pHistTrueFlux->Draw("hist");
186  PlottingHelper::SaveCanvas(pCanvas, "plotFluxVariations_trueFlux_" + paramName);
187 
188  pHistFracDiff->Draw("hist");
189  PlottingHelper::SaveCanvas(pCanvas, "plotFluxVariations_fracDiff_" + paramName);
190 
191  pHist2D->Draw("colz");
192  PlottingHelper::SaveCanvas(pCanvas, "plotFluxVariations_fracDiff-vs-trueFlux_" + paramName);
193 
194  table.AddEmptyRow();
195  table.SetEntry("Parameter", paramName);
196  table.SetEntry("True flux STD", pHistTrueFlux->GetStdDev());
197  table.SetEntry("Frac diff RMS", pHistFracDiff->GetRMS());
198  }
199 
200  table.WriteToFile("plotFluxVariations_differences.md");
201 }
