80 
81  //// BEGIN TEST
82  if (isTrueSignal && config.makeSelectedPIDTable.onlyLowMomentumPions)
83  {
84  // Get the true pion momentum
85  const auto truthData = AnalysisHelper::GetTruthAnalysisData(pEvent->truth, config.global.useAbsPdg, config.global.protonMomentumThreshold);
86 
87  // Check if this pion has sufficiently low momentum
88  isTrueSignal = (truthData.pionMomentum < config.makeSelectedPIDTable.pionMomentumThreshold);
89  }
90  //// END TEST
91 
92  // Run the event selection and store which cuts are passed
93  const auto &[isSelectedTotal, cutsPassed, assignedPdgCodes] = selection.Execute(pEvent);
94  const auto isSelectedAtChosenCut = (std::find(cutsPassed.begin(), cutsPassed.end(), lastCut) != cutsPassed.end());
95 
96  // Don't bother with events that weren't selected
97  if (!isSelectedAtChosenCut)
98  continue;
99 
100  if (assignedPdgCodes.size() != recoParticles.size())
101  throw std::logic_error("MakeSelectedPIDTable - The output particle PDG codes is the wrong size");
102 
103  for (unsigned int index = 0; index < recoParticles.size(); ++index)
104  {
105  const auto &particle = recoParticles.at(index);
106 
107  // Get the true PDG code (using 0 for EXTs)
108  int truePdgCode = 0;
109  if (isOverlay || isDirt)
110  {
111  try
112  {
113  const auto truthParticle = AnalysisHelper::GetBestMatchedTruthParticle(particle, truthParticles, true);
114  truePdgCode = config.global.useAbsPdg ? std::abs(truthParticle.pdgCode()) : truthParticle.pdgCode();
115  }
116  catch (std::exception &)
117  {
118  }
119  }
120 
121  const auto recoPdgCode = assignedPdgCodes.at(index);
122  if (recoPdgCode != 13 && recoPdgCode != 211 && recoPdgCode != 2212)
123  throw std::logic_error("MakeSelectedPIDTable - Unknown assigned PDG code of " + std::to_string(recoPdgCode));
124 
125  // Get the mapping from true PDG to count for this reco pdg code, making it if it doesn't exist
126  auto &truePdgToCountMap = recoToTruePdgMap[recoPdgCode];
127 
128  // Increment the count
129  auto iter = truePdgToCountMap.find(truePdgCode);
130  if (iter == truePdgToCountMap.end())
131  {
132  std::unordered_map<bool, float> newEntry;
133  newEntry.emplace(true, isTrueSignal ? weight : 0.f); // Only count "true" key for signal events
134  newEntry.emplace(false, weight); // Count "false" key for all events
135 
136  truePdgToCountMap.emplace(truePdgCode, newEntry);
137  }
138  else
139  {
140  if (isTrueSignal)
141  iter->second.at(true) += weight;
142 
143  iter->second.at(false) += weight;
144  }
145 
146  // Do the same for dirt events only
147  if (sampleType == AnalysisHelper::Dirt)
148  {
149  auto &truePdgToCountMapDirt = recoToTruePdgMapDirt[recoPdgCode];
150  auto iterDirt = truePdgToCountMapDirt.find(truePdgCode);
151  if (iterDirt == truePdgToCountMapDirt.end())
152  {
153  truePdgToCountMapDirt.emplace(truePdgCode, weight);
154  }
155  else
156  {
157  iterDirt->second += weight;
158  }
159  }
160  }
161  }
162  }
163 
164  // Extract the PDG codes we want to print
165  std::vector<int> recoPdgs, truePdgs;
166  std::unordered_map<int, float> recoPdgToTotalWeightMap, recoPdgToTotalWeightMapSignal, recoPdgToTotalWeightMapDirt;
167 
168  for (const auto &recoEntry : recoToTruePdgMap)
169  {
170  const auto &recoPdgCode = recoEntry.first;
171 
172  // Store this PDG code if not yet seen
173  if (std::find(recoPdgs.begin(), recoPdgs.end(), recoPdgCode) == recoPdgs.end())
174  {
175  recoPdgs.push_back(recoPdgCode);
176  recoPdgToTotalWeightMap.emplace(recoPdgCode, 0.f);
177  recoPdgToTotalWeightMapSignal.emplace(recoPdgCode, 0.f);
178  recoPdgToTotalWeightMapDirt.emplace(recoPdgCode, 0.f);
179  }
180 
181  for (const auto &trueEntry : recoEntry.second)
182  {
183  const auto &truePdgCode = trueEntry.first;
184 
185  recoPdgToTotalWeightMap.at(recoPdgCode) += trueEntry.second.at(false);
186  recoPdgToTotalWeightMapSignal.at(recoPdgCode) += trueEntry.second.at(true);
187 
188  // Store this PDG code if not yet seen
189  if (std::find(truePdgs.begin(), truePdgs.end(), truePdgCode) == truePdgs.end())
190  truePdgs.push_back(truePdgCode);
191  }
192  }
193 
194  for (const auto &recoEntry : recoToTruePdgMapDirt)
195  {
196  const auto &recoPdgCode = recoEntry.first;
197 
198  // Store this PDG code if not yet seen
199  if (std::find(recoPdgs.begin(), recoPdgs.end(), recoPdgCode) == recoPdgs.end())
200  {
201  recoPdgs.push_back(recoPdgCode);
202  recoPdgToTotalWeightMap.emplace(recoPdgCode, 0.f);
203  recoPdgToTotalWeightMapSignal.emplace(recoPdgCode, 0.f);
204  recoPdgToTotalWeightMapDirt.emplace(recoPdgCode, 0.f);
205  }
206 
207  for (const auto &trueEntry : recoEntry.second)
208  {
209  const auto &truePdgCode = trueEntry.first;
210  const auto weight = trueEntry.second;
211 
212  recoPdgToTotalWeightMapDirt.at(recoPdgCode) += weight;
213 
214  // Store this PDG code if not yet seen
215  if (std::find(truePdgs.begin(), truePdgs.end(), truePdgCode) == truePdgs.end())
216  truePdgs.push_back(truePdgCode);
217  }
218  }
219 
220  std::sort(recoPdgs.begin(), recoPdgs.end());
221  std::sort(truePdgs.begin(), truePdgs.end());
222 
223  // Now fill the output tables
224  auto tableHeaders = std::vector<std::string>({"True PDG", ""});
225  for (const auto &pdg : recoPdgs)
226  tableHeaders.push_back(std::to_string(pdg));
227 
228  FormattingHelper::Table table(tableHeaders);

230  FormattingHelper::Table dirtTable(tableHeaders);
231 
232  // Add the sum row
233  table.AddEmptyRow();
234  signalTable.AddEmptyRow();

236 
237  table.SetEntry("True PDG", "all");
238  signalTable.SetEntry("True PDG", "all");
239  dirtTable.SetEntry("True PDG", "all");
240 
241  for (const auto &recoPdg : recoPdgs)
242  {
243  const auto count = recoPdgToTotalWeightMap.at(recoPdg);
244  const auto signalCount = recoPdgToTotalWeightMapSignal.at(recoPdg);
245  const auto dirtCount = recoPdgToTotalWeightMapDirt.at(recoPdg);
246 
247  table.SetEntry(std::to_string(recoPdg), count);
248  signalTable.SetEntry(std::to_string(recoPdg), signalCount);
249  dirtTable.SetEntry(std::to_string(recoPdg), dirtCount);
250  }
251 
252  for (const auto &truePdg : truePdgs)
253  {
254  table.AddEmptyRow();
255  signalTable.AddEmptyRow();
256  dirtTable.AddEmptyRow();
257 
258  table.SetEntry("True PDG", truePdg);
259  signalTable.SetEntry("True PDG", truePdg);
260  dirtTable.SetEntry("True PDG", truePdg);
261 
262  for (const auto &recoPdg : recoPdgs)
263  {
264  const auto countTotal = recoPdgToTotalWeightMap.at(recoPdg);
265  const auto signalCountTotal = recoPdgToTotalWeightMapSignal.at(recoPdg);
266  const auto dirtCountTotal = recoPdgToTotalWeightMapDirt.at(recoPdg);
267 
268  float count = 0.f;
269  float signalCount = 0.f;
270  float dirtCount = 0.f;
271 
272  const auto &entry = recoToTruePdgMap.at(recoPdg);
273  const auto iter = entry.find(truePdg);
274 
275  if (iter != entry.end())
276  {
277  count = iter->second.at(false);
278  signalCount = iter->second.at(true);
279  }
280 
281  table.SetEntry(std::to_string(recoPdg), count / countTotal);
282  signalTable.SetEntry(std::to_string(recoPdg), signalCount / signalCountTotal);
283 
284  const auto &entryDirt = recoToTruePdgMapDirt.at(recoPdg);
285  const auto iterDirt = entryDirt.find(truePdg);
286 
287  if (iterDirt != entryDirt.end())
288  {
289  dirtCount = iterDirt->second;
290  }
291 
292  dirtTable.SetEntry(std::to_string(recoPdg), dirtCount / dirtCountTotal);
293  }
294  }
295 
296  // Print the tables
297  const auto prefix = std::string("pidTable") + (config.makeSelectedPIDTable.goldenPionIsSignal ? "_goldenPionIsSignal" : "") + std::string("_atCut-") + lastCut;
298 
299  FormattingHelper::PrintLine();
300  std::cout << "All events" << std::endl;
301  FormattingHelper::PrintLine();
302  table.WriteToFile(prefix + "_allEvents.md");
303 
304  std::cout << std::endl;
305  FormattingHelper::PrintLine();
306  std::cout << "Signal events" << std::endl;
307  FormattingHelper::PrintLine();
308  signalTable.WriteToFile(prefix + "_signalEvents.md");
309 
310  std::cout << std::endl;
311  FormattingHelper::PrintLine();
312  std::cout << "Dirt events" << std::endl;
313  FormattingHelper::PrintLine();
314  dirtTable.WriteToFile(prefix + "_dirtEvents.md");
315 }
