923 
924 void AnalysisHelper::GetRangeToMomentumFitParameters(const std::vector<float> &ranges, const std::vector<float> &momenta, RangeToMomentumFitParameters &params)
925 {
926  if (ranges.size() != momenta.size())
927  throw std::invalid_argument("AnalysisHelper::GetRangeToMomentumFitParameters - input ranges and momenta are of different size");
928 
929  // In general for a given range, there can be a significant (and asymmetric) spread of momenta around the most probable value.
930  // The idea here is to first bin the ranges, and get the most probable value at each range, then fit those most probable values
931 
932  // The number of data points to put in each bin
933  const unsigned int nPointsPerBin = 100u;
934 
935  // The maximum allowed bin width
936  const float maxBinWidth = 10.f;
937 
938  // The momentum resolution scale to use when finding the MPV
939  const float momentumScale = 0.01f;
940 
941  // First turn the two input vectors into a vector of pairs so they can be sorted
942  std::vector< std::pair<float, float> > dataPoints;
943  for (unsigned int i = 0; i < ranges.size(); ++i)
944  {
945  dataPoints.emplace_back(ranges.at(i), momenta.at(i));
946  }
947 
948  // Now sort the points by range
949  std::sort(dataPoints.begin(), dataPoints.end(), [](const auto &a, const auto &b){ return a.first < b.first; });
950 
951  // Now organise the data points into bins, each bin corresponds to a set of nearby ranges
952  std::vector< std::vector< std::pair<float, float> > > binnedPoints;
953  for (unsigned int i = 0; i < dataPoints.size(); ++i)
954  {
955  // If we have enough points in the current bin, then add another
956  if (i % nPointsPerBin == 0)
957  binnedPoints.emplace_back();
958 
959  // Add this data point to the last bin
960  binnedPoints.back().push_back(dataPoints.at(i));
961  }
962 
963  // Next find the most probable value in each bin
964  std::vector<float> fitRanges, fitMomenta;
965  for (const auto &bin : binnedPoints)
966  {
967  // Ignore the last bin if it doesn't have enough points
968  if (bin.size() < nPointsPerBin)
969  continue;
970 
971  // Extract the ranges and momenta in this bin
972  std::vector<float> rangesInBin, momentaInBin;
973  for (const auto &point : bin)
974  {
975  rangesInBin.push_back(point.first);
976  momentaInBin.push_back(point.second);
977  }
978 
979  // Get the spread of ranges in the bin
980  const auto minRange = *std::min_element(rangesInBin.begin(), rangesInBin.end());
981  const auto maxRange = *std::max_element(rangesInBin.begin(), rangesInBin.end());
982  const auto binWidth = maxRange - minRange;
983 
984  // Don't use bins that have a large spread of values
985  if (binWidth > maxBinWidth)
986  continue;
987 
988  // Get the mean of the ranges in the bin
989  const auto meanRange = std::accumulate(rangesInBin.begin(), rangesInBin.end(), 0.f) / static_cast<float>(rangesInBin.size());
990 
991  // Get the minimum momenta in the bin
992  const auto minMomentum = *std::min_element(momentaInBin.begin(), momentaInBin.end());
993 
994  // Organise the momenta into their own bins
995  std::map<unsigned int, std::vector<float> > momentumBins;
996  for (const auto &momentum : momentaInBin)
997  {
998  const unsigned int binIndex = std::floor((momentum - minMomentum) / momentumScale);
999  momentumBins[binIndex].push_back(momentum);
1000  }
1001 
1002  // Sort the momentum bins by the number of entries
1003  std::vector< std::pair<unsigned int, std::vector<float> > > momentumBinsVector;
1004  for (const auto &momentumBin : momentumBins)
1005  {
1006  momentumBinsVector.emplace_back(momentumBin.first, momentumBin.second);
1007  }
1008  std::sort(momentumBinsVector.begin(), momentumBinsVector.end(), [](const auto &a, const auto &b){ return a.second.size() > b.second.size(); });
1009 
1010  // Find the mean momenta in the most probable bin
1011  const auto mostProbableMomenta = momentumBinsVector.front().second;
1012  const auto meanMomentum = std::accumulate(mostProbableMomenta.begin(), mostProbableMomenta.end(), 0.f) / static_cast<float>(mostProbableMomenta.size());
1013 
1014  // Save this as a data point to fit
1015  fitRanges.push_back(meanRange);
1016  fitMomenta.push_back(meanMomentum);
1017  }
1018 
1019  // Determine the range over which to fit
1020  const auto minRange = *std::min_element(fitRanges.begin(), fitRanges.end());
1021  const auto maxRange = *std::max_element(fitRanges.begin(), fitRanges.end());
1022 
1023 
1024  // Make a TGraph to fit
1025  const auto nPoints = fitRanges.size();
1026  std::shared_ptr<TGraph> pGraph(new TGraph(nPoints, fitRanges.data(), fitMomenta.data()));
1027 
1028  // Get the function we want to fit
1029  auto pFunc = AnalysisHelper::GetRangeToMomentumFunction();
1030 
1031  // Set some initial values to help the fit converge quickly. For reference, I just eyeballed these numbers from a plot of muons
1032  params.a = 2e-1f;
1033  params.b = 2e-3f;
1034  params.c = 1e-1f;
1035  params.d = 2e-1f;
1036  AnalysisHelper::SetRangeToMomentumFunctionParameters(params, pFunc);
1037 
1038  // Run the fit
1039  // - M = do it gooder (about as much information as the ROOT documentation gives you!)
1040  // - N = don't draw the result
1041  // - Q = quiet mode
1042  // - S = return the fit result
1043  const auto fitResult = pGraph->Fit(pFunc->GetName(), "MNQS", "", minRange, maxRange);
1044  if (!fitResult->IsValid())
1045  std::cout << "AnalysisHelper::GetRangeToMomentumFitParameters - WARNING. Fit was deemed problematic, check the result!" << std::endl;
1046 
1047  // Store the fitted parameters
1048  AnalysisHelper::GetRangeToMomentumFunctionParameters(pFunc, params);
1049 }
