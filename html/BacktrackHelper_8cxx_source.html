91 HitsToBool BacktrackHelper::GetHitsToIsNuInducedMap(const art::Event &event, const art::InputTag &hitLabel, const art::InputTag &mcParticleLabel, const art::InputTag &backtrackerLabel, const MCParticleVector &nuMCParticles)
92 {

94  const auto mcpToHitMap = CollectionHelper::GetAssociation<simb::MCParticle, recob::Hit>(event, mcParticleLabel, backtrackerLabel);
95  const auto hitToMcpMap = CollectionHelper::GetReversedAssociation(mcpToHitMap);
96 
97  HitsToBool outputMap;
98 
99  for (const auto &hit : hits)
100  {
101  // Start by assuming that the hit isn't neutrino induced, and change this later if required
102  if (!outputMap.emplace(hit, false).second)
103  throw cet::exception("BacktrackHelper::GetHitsToIsNuInducedMap") << " - Repeated input hits." << std::endl;
104 
105  // Check the hit is associated to an MCParticle
106  const auto iter = hitToMcpMap.find(hit);
107  if (iter == hitToMcpMap.end())

111  bool isNuInduced = false;

113  {
114  if (std::find(nuMCParticles.begin(), nuMCParticles.end(), mcParticle) != nuMCParticles.end())
115  {
116  isNuInduced = true;

118  }
119  }
120 
121  if (isNuInduced)
122  outputMap.at(hit) = true;
123  }

722 art::Ptr<recob::Slice> BacktrackHelper::SliceMetadata::GetMostCompleteSlice() const
723 {
724  if (this->GetTotalNumberOfNuInducedHits() == 0)
725  throw cet::exception("SliceMetadata::GetMostCompleteSlice") << " - there are no neutrino hits! No slice is more complete than any other" << std::endl;
726 
727  if (m_slices.empty())
728  throw cet::exception("SliceMetadata::GetMostCompleteSlice") << " - there are no slices to choose from." << std::endl;
729 
730  float bestCompleteness = -std::numeric_limits<float>::max();
731  art::Ptr<recob::Slice> mostCompleteSlice;
732  bool foundMostCompleteSlice = false;
733 
734  for (const auto &slice : m_slices)
735  {
736  const auto completeness = this->GetCompleteness(slice);
737  if (completeness < bestCompleteness)
738  continue;
739 
740  bestCompleteness = completeness;
741  mostCompleteSlice = slice;
742  foundMostCompleteSlice = true;
743  }
744 
745  if (!foundMostCompleteSlice)
746  throw cet::exception("SliceMetadata::GetMostCompleteSlice") << " - couldn't find the most complete slice" << std::endl;
747 
748  return mostCompleteSlice;
749 }
